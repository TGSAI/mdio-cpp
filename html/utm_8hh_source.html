82 inline GeoCoord utm_to_geo(const UTMCoord& utm) {
83  // Constants for GDA94 / MGA Zone 51 (EPSG:28351)
84  const double a = 6378137.0; // Semi-major axis
85  const double f = 1.0 / 298.257222101; // Flattening
86  const double k0 = 0.9996; // Scale factor
87  const double central_meridian = 123.0; // Central meridian for Zone 51
88  const double false_easting = 500000.0; // False easting
89  const double false_northing = 10000000.0; // False northing (Southern Hemisphere)
90 
91  // Calculate ellipsoid parameters
92  const double e2 = 2 * f - f * f; // Eccentricity squared
93  const double e_prime_squared = e2 / (1 - e2);
94 
95  // Adjust for false easting and northing
96  const double x = utm.easting - false_easting;
97  const double y = utm.northing - false_northing;
98 
99  // Meridian distance
100  const double m = y / k0;
101 
102  // Footprint latitude
103  const double mu = m / (a * (1 - e2/4 - 3*e2*e2/64 - 5*e2*e2*e2/256));
104 
105  // Coefficients for footprint latitude
106  const double e1 = (1 - sqrt(1 - e2)) / (1 + sqrt(1 - e2));
107  const double j1 = 3*e1/2 - 27*e1*e1*e1/32;
108  const double j2 = 21*e1*e1/16 - 55*e1*e1*e1*e1/32;
109  const double j3 = 151*e1*e1*e1/96;
110  const double j4 = 1097*e1*e1*e1*e1/512;
111 
112  // Footprint latitude
113  const double fp = mu + j1*sin(2*mu) + j2*sin(4*mu) + j3*sin(6*mu) + j4*sin(8*mu);
114 
115  // Calculate parameters
116  const double cos_fp = cos(fp);
117  const double sin_fp = sin(fp);
118  const double tan_fp = tan(fp);
119 
120  const double C1 = e_prime_squared * cos_fp * cos_fp;
121  const double T1 = tan_fp * tan_fp;
122  const double R1 = a * (1 - e2) / pow(1 - e2 * sin_fp * sin_fp, 1.5);
123  const double N1 = a / sqrt(1 - e2 * sin_fp * sin_fp);
124 
125  const double D = x / (N1 * k0);
126 
127  // Calculate latitude
128  const double lat_rad = fp - (tan_fp / (R1 * N1)) * (
129  D*D/2 -
130  (5 + 3*T1 + 10*C1 - 4*C1*C1 - 9*e_prime_squared) * D*D*D*D/24 +
131  (61 + 90*T1 + 298*C1 + 45*T1*T1 - 252*e_prime_squared - 3*C1*C1) * D*D*D*D*D*D/720
132  );
133 
134  // Calculate longitude
135  const double lon_rad = central_meridian * M_PI / 180.0 + (
136  D -
137  (1 + 2*T1 + C1) * D*D*D/6 +
138  (5 - 2*C1 + 28*T1 - 3*C1*C1 + 8*e_prime_squared + 24*T1*T1) * D*D*D*D*D/120
139  ) / cos_fp;
140 
141  // Convert to degrees
142  GeoCoord result;
143  result.latitude = lat_rad * 180.0 / M_PI;
144  result.longitude = lon_rad * 180.0 / M_PI;
145 
146  return result;
147 }

156 std::string urlEncode(const std::string &value) {
157  std::ostringstream escaped;
158  escaped.fill('0');
159  escaped << std::hex;
160 
161  for (unsigned char c : value) {
162  // Encode reserved characters
163  if (isalnum(c) || c == '-' || c == '_' || c == '.' || c == '~') {
164  escaped << c;
165  } else {
166  escaped << '%' << std::uppercase << std::setw(2) << int(c);
167  }
168  }
169  return escaped.str();
170 }

194 std::string encodeGeoJSONBoundingBoxURL(const CoordExtent& x_extents, const CoordExtent& y_extents) {
195  // Calculate the four corners in UTM coordinates
196  UTMCoord nw = {x_extents.min, y_extents.max};
197  UTMCoord ne = {x_extents.max, y_extents.max};
198  UTMCoord sw = {x_extents.min, y_extents.min};
199  UTMCoord se = {x_extents.max, y_extents.min};
200 
201  // Convert to geographic coordinates
202  GeoCoord nw_geo = utm_to_geo(nw);
203  GeoCoord ne_geo = utm_to_geo(ne);
204  GeoCoord sw_geo = utm_to_geo(sw);
205  GeoCoord se_geo = utm_to_geo(se);
206 
207  std::ostringstream geojson;
208  geojson << R"({"type":"Polygon","coordinates":[[)"
209  << "[" << nw_geo.longitude << "," << nw_geo.latitude << "],"
210  << "[" << ne_geo.longitude << "," << ne_geo.latitude << "],"
211  << "[" << se_geo.longitude << "," << se_geo.latitude << "],"
212  << "[" << sw_geo.longitude << "," << sw_geo.latitude << "],"
213  << "[" << nw_geo.longitude << "," << nw_geo.latitude << "]]]}"; // Close the polygon
214 
215  std::string geojsonStr = geojson.str();
216  std::string encodedGeoJSON = urlEncode(geojsonStr);
217 
218  return "http://geojson.io/#data=data:application/json," + encodedGeoJSON;
219 }

228 void print_corners(const CoordExtent& x_extents, const CoordExtent& y_extents) {
229  // Display the four corners of the geometry
230  std::cout << "\nGeometry Corners (UTM -> Lat/Long):" << std::endl;
231  std::cout << "-------------------------------------" << std::endl;
232 
233  // Check if we're dealing with a point (both extents are points with the same value)
234  if (x_extents.is_point() && y_extents.is_point()) {
235  GeoCoord geo = utm_to_geo({x_extents.min, y_extents.min});
236  std::cout << "Point: E=" << x_extents.min << ", N=" << y_extents.min
237  << " -> Lat=" << geo.latitude << "°, Lon=" << geo.longitude << "°" << std::endl;
238  return;
239  }
240 
241  // Northwest corner
242  UTMCoord nw = {x_extents.min, y_extents.max};
243  GeoCoord nw_geo = utm_to_geo(nw);
244  std::cout << "Northwest: E=" << nw.easting << ", N=" << nw.northing
245  << " -> Lat=" << nw_geo.latitude << "°, Lon=" << nw_geo.longitude << "°" << std::endl;
246 
247  // Northeast corner
248  UTMCoord ne = {x_extents.max, y_extents.max};
249  GeoCoord ne_geo = utm_to_geo(ne);
250  std::cout << "Northeast: E=" << ne.easting << ", N=" << ne.northing
251  << " -> Lat=" << ne_geo.latitude << "°, Lon=" << ne_geo.longitude << "°" << std::endl;
252 
253  // Southwest corner
254  UTMCoord sw = {x_extents.min, y_extents.min};
255  GeoCoord sw_geo = utm_to_geo(sw);
256  std::cout << "Southwest: E=" << sw.easting << ", N=" << sw.northing
257  << " -> Lat=" << sw_geo.latitude << "°, Lon=" << sw_geo.longitude << "°" << std::endl;

259  // Southeast corner
