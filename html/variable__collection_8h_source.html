66  template <typename T = void, DimensionIndex R = dynamic_rank,
67  ReadWriteMode M = ReadWriteMode::dynamic>
68  Result<Variable<T, R, M>> get(const std::string& label) const {
69  if (!variables.count(label)) {
70  return absl::NotFoundError("Label '" + label +
71  "' not found in the stores map");
72  }
73 
74  auto cast_store =
75  tensorstore::StaticCast<tensorstore::TensorStore<T, R, M>>(
76  variables.at(label).get_store());
77 
78  if (!cast_store.ok()) {
79  return cast_store.status();
80  }
81 
82  return Variable<T, R, M>{variables.at(label).get_variable_name(),
83  variables.at(label).get_long_name(),
84  variables.at(label).getReducedMetadata(),
85  cast_store.value(),
86  variables.at(label).attributes};
87  }

148  std::vector<std::string> get_iterable_accessor() const {
149  // The intention here is to hide the std iterators from the user.
150  // unordered_map iterators are not guaranteed to have the same order across
151  // different compilers.
152  std::vector keys = get_keys();
153  std::sort(keys.begin(), keys.end());
154  return keys;
155  }
