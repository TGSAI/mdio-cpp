57 template <typename T>
58 mdio::Result<void> populate_and_write_variable(
59  mdio::Dataset& dataset, const std::string& variable_name,
60  std::function<void(SharedArray<T>&)> populate_func) {
61  MDIO_ASSIGN_OR_RETURN(auto variable_data,
62  from_dataset<T>(dataset, variable_name));
63  auto data = variable_data.get_data_accessor();
64 
65  populate_func(data);
66 
67  MDIO_ASSIGN_OR_RETURN(auto variable, dataset.variables.get<T>(variable_name));
68 
69  return variable.Write(variable_data).result();
70 }
