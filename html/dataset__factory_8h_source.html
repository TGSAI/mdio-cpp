53 tensorstore::Result<std::string> to_zarr_dtype(const std::string dtype) {
54  // Convert the input dtype to Zarr dtype
55  if (dtype == "int8") {
56  return "<i1";
57  } else if (dtype == "int16") {
58  return "<i2";
59  } else if (dtype == "int32") {
60  return "<i4";
61  } else if (dtype == "int64") {
62  return "<i8";
63  } else if (dtype == "uint8") {
64  return "<u1";
65  } else if (dtype == "uint16") {
66  return "<u2";
67  } else if (dtype == "uint32") {
68  return "<u4";
69  } else if (dtype == "uint64") {
70  return "<u8";
71  } else if (dtype == "float16") {
72  return "<f2";
73  } else if (dtype == "float32") {
74  return "<f4";
75  } else if (dtype == "float64") {
76  return "<f8";
77  } else if (dtype == "bool") {
78  return "|b1";
79  } else if (dtype == "complex64") {
80  return "<c8";
81  } else if (dtype == "complex128") {
82  return "<c16";
83  }
84  return absl::InvalidArgumentError("Unknown dtype: " + dtype);
85 }

96 absl::Status transform_dtype(nlohmann::json& input /*NOLINT*/,
97  nlohmann::json& variable /*NOLINT*/) {
98  if (input["dataType"].contains("fields")) {
99  nlohmann::json dtypeFields = nlohmann::json::array();
100  for (const auto& field : input["dataType"]["fields"]) {
101  auto dtype = to_zarr_dtype(field["format"]);
102  if (!dtype.status().ok()) {
103  return dtype.status();
104  }
105  dtypeFields.emplace_back(nlohmann::json{field["name"], dtype.value()});
106  }
107  variable["metadata"]["dtype"] = dtypeFields;
108  } else {
109  auto dtype = to_zarr_dtype(input["dataType"]);
110  if (!dtype.status().ok()) {
111  return dtype.status();
112  }
113  variable["metadata"]["dtype"] = dtype.value();
114  }
115  return absl::OkStatus();
116 }

127 absl::Status transform_compressor(nlohmann::json& input /*NOLINT*/,
128  nlohmann::json& variable /*NOLINT*/) {
129  if (input.contains("compressor")) {
130  if (input["compressor"].contains("name")) {
131  if (input["compressor"]["name"] != "blosc") {
132  return absl::InvalidArgumentError("Only blosc compressor is supported");
133  }
134  variable["metadata"]["compressor"]["id"] = input["compressor"]["name"];
135  } else {
136  return absl::InvalidArgumentError("Compressor name must be specified");
137  }
138 
139  if (input["compressor"].contains("algorithm")) {
140  variable["metadata"]["compressor"]["cname"] =
141  input["compressor"]["algorithm"];
142  } else { // DEFAULT
143  variable["metadata"]["compressor"]["cname"] = "lz4";
144  }
145  if (input["compressor"].contains("level")) {
146  if (input["compressor"]["level"] > 9 ||
147  input["compressor"]["level"] < 0) {
148  return absl::InvalidArgumentError(
149  "Compressor level must be between 0 and 9");
150  }
151  variable["metadata"]["compressor"]["clevel"] =
152  input["compressor"]["level"];
153  } else { // DEFAULT
154  variable["metadata"]["compressor"]["clevel"] = 5;
155  }
156  if (input["compressor"].contains("shuffle")) {
157  variable["metadata"]["compressor"]["shuffle"] =
158  input["compressor"]["shuffle"];
159  } else { // DEFAULT
160  variable["metadata"]["compressor"]["shuffle"] = 1;
161  }
162  if (input["compressor"].contains("blocksize")) {
163  variable["metadata"]["compressor"]["blocksize"] =
164  input["compressor"]["blocksize"];
165  } else { // DEFAULT
166  variable["metadata"]["compressor"]["blocksize"] = 0;
167  }
168  } else {
169  variable["metadata"]["compressor"] = nullptr;
170  }
171  return absl::OkStatus();
172 }

211 absl::Status transform_metadata(const std::string& path,
212  nlohmann::json& variable /*NOLINT*/) {
213  std::string bucket =
214  "NULL"; // Default value, if is NULL don't add a bucket field
215  std::string driver = "file";
216  if (absl::StartsWith(path, "gs://")) {
217  driver = "gcs";
218  } else if (absl::StartsWith(path, "s3://")) {
219  driver = "s3";
220  }
221 
222  std::string filepath;
223 
224  if (driver != "file") {
225  std::string _path = path;
226  // Ensure _path has a trailing slash or the mdio file will not be created
227  // properly
228  if (_path.back() != '/') {
229  _path += '/';
230  }
231  _path = _path.substr(5);
232  std::vector<std::string> file_parts = absl::StrSplit(_path, '/');
233  if (file_parts.size() < 2) {
234  return absl::InvalidArgumentError(
235  "Cloud path requires [gs/s3]://[bucket]/[path to file] name");
236  }
237  bucket = file_parts[0];
238  filepath = file_parts[1];
239  for (std::size_t i = 2; i < file_parts.size(); ++i) {
240  filepath += "/" + file_parts[i];
241  }
242  filepath += variable["kvstore"]["path"].get<std::string>();
243  } else {
244  filepath = path + "/";
245  filepath += variable["kvstore"]["path"].get<std::string>();
246  }
247  variable["kvstore"]["path"] = filepath;
248  variable["kvstore"]["driver"] = driver;
249  if (bucket != "NULL") {
250  variable["kvstore"]["bucket"] = bucket;
251  }
252 
253  return absl::OkStatus();
254 }

264 tensorstore::Result<nlohmann::json> from_json_to_spec(
265  nlohmann::json& json /*NOLINT*/,
266  std::unordered_map<std::string, uint64_t>& dimensionMap /*NOLINT*/,
267  const std::string& path) {
268  nlohmann::json variableStub = R"(
269  {
270  "driver": "zarr",
271  "kvstore": {
272  "driver": "file",
273  "path": "VARIABLE_NAME"
274  },
275  "metadata": {
276  "dtype": "DATA_TYPE",
277  "dimension_separator": "/",
278  "shape": "SHAPE",
279  "chunks": "CHUNKS"
280  },
281  "attributes": {}
282  }
283  )"_json;
284  variableStub["kvstore"]["path"] = json["name"];
285 
286  auto transformStatus = transform_dtype(json, variableStub);
287  if (!transformStatus.ok()) {
288  return transformStatus;
289  }
290 
291  auto compressorStatus = transform_compressor(json, variableStub);
292  if (!compressorStatus.ok()) {
293  return compressorStatus;
294  }
295 
296  transform_shape(json, variableStub, dimensionMap);
297 
298  if (json.contains("metadata")) {
299  if (json["metadata"].contains("chunkGrid")) {
300  variableStub["metadata"]["chunks"] =
301  json["metadata"]["chunkGrid"]["configuration"]["chunkShape"];
302  } else { // No chunking specified
303  variableStub["metadata"]["chunks"] = variableStub["metadata"]["shape"];
304  }
305 
306  if (!json["dataType"].contains("fields")) {
307  variableStub["metadata"]["fill_value"] = nlohmann::json::value_t::null;
308  if (json["dataType"] == "complex64") {
309  std::string raw(8, '\0');
310  variableStub["metadata"]["fill_value"] = encode_base64(raw);
311  } else if (json["dataType"] == "complex128") {
312  std::string raw(16, '\0');
313  variableStub["metadata"]["fill_value"] = encode_base64(raw);
314  } else if (json["dataType"].get<std::string>()[0] == 'f') {
315  variableStub["metadata"]["fill_value"] = std::nan("");
316  }
317  } else {
318  // Accumulate the total number of bytes (N)
319  uint16_t num_bytes = 0;
320  // We're going to use the variable dtype because it's already in byte
321  // format
322  std::string dtype;
323  for (auto field : variableStub["metadata"]["dtype"]) {
324  dtype = field[1].get<std::string>();
325  if (dtype.at(1) != 'c') {
326  num_bytes += std::stoi(dtype.substr(2));
327  } else { // Complex
328  if (dtype.at(2) == '8') {
329  num_bytes += 8;
330  } else {
331  num_bytes += 16;
332  }
333  }
334  }
335  std::string raw(num_bytes, '\0');
336  variableStub["metadata"]["fill_value"] = encode_base64(raw);
337  }
338 
339  variableStub["attributes"]["metadata"] = json["metadata"];
340  } else { // No metadata supplied means no chunkGrid
341  variableStub["metadata"]["chunks"] = variableStub["metadata"]["shape"];
342  }
343 
344  auto transform_result = transform_metadata(path, variableStub);
345  if (!transform_result.ok()) {
346  return transform_result;
347  }
348 
349  // I think the longName field should be optional for a Variable but this
350  // ensures the spec is valid.
351  if (json.contains("longName")) {
352  variableStub["attributes"]["long_name"] = json["longName"];
353  } else {
354  variableStub["attributes"]["long_name"] = "";
355  }
356 
357  if (!json.contains("dimensions")) {
358  nlohmann::json dimension_names = nlohmann::json::array();
359  dimension_names.emplace_back(json["name"]);
360  variableStub["attributes"]["dimension_names"] = dimension_names;
361  } else if (json["dimensions"][0].is_object()) {
362  nlohmann::json dimension_names = nlohmann::json::array();
363  for (size_t i = 0; i < json["dimensions"].size(); ++i) {
364  dimension_names.emplace_back(json["dimensions"][i]["name"]);
365  }
366  variableStub["attributes"]["dimension_names"] = dimension_names;
367  } else {
368  variableStub["attributes"]["dimension_names"] = json["dimensions"];
369  }
370 
371  // We do not want to seralize "dimension coordinates"
372  std::set<std::string> dims;
373  for (auto& dim : variableStub["attributes"]["dimension_names"]) {
374  dims.insert(dim.get<std::string>());
375  }
376 
377  if (json.contains("coordinates")) {
378  // This appears to need to be a space separated string, not a list
379  std::string coordinates;
380  std::size_t coords_size = json["coordinates"].size();
381  for (size_t i = 0; i < coords_size; ++i) {
382  std::string coord = json["coordinates"][i].get<std::string>();
383  if (dims.count(coord) > 0) {
384  continue;
385  }
386  coordinates += coord;
387  if (i != coords_size - 1) {
388  coordinates += " ";
389  }
390  }
391  variableStub["attributes"]["coordinates"] = coordinates;
392  }
393 
394  return variableStub;
395 }

407 tensorstore::Result<std::unordered_map<std::string, uint64_t>> get_dimensions(
408  nlohmann::json& spec /*NOLINT*/) {
409  std::unordered_map<std::string, uint64_t> dimensions;
410  for (auto& variable : spec["variables"]) {
411  if (variable["dimensions"][0].is_object()) {
412  for (auto& dimension : variable["dimensions"]) {
413  if (dimension["size"].get<uint64_t>() > mdio::constants::kMaxSize) {
414  return absl::InvalidArgumentError(
415  "Dimension " + dimension["name"].dump() +
416  " exceeds maximum size of " +
417  std::to_string(mdio::constants::kMaxSize));
418  }
419  if (dimensions.count(dimension["name"]) == 0) {
420  dimensions[dimension["name"]] = dimension["size"];
421  } else {
422  if (dimensions[dimension["name"]] != dimension["size"]) {
423  return absl::InvalidArgumentError("Dimension " +
424  dimension["name"].dump() +
425  " has conflicting sizes");
426  }
427  }
428  }
429  }
430  }
431  return dimensions;
432 }

441 tensorstore::Result<std::tuple<nlohmann::json, std::vector<nlohmann::json>>>
442 Construct(nlohmann::json& spec /*NOLINT*/, const std::string& path) {
443  // Validation should only return status codes. If it returns data then it
444  // should be a "constructor"
445  auto status = validate_dataset(spec);
446  if (!status.ok()) {
447  return status;
448  }
449 
450  // This made more sense to validate in the constructor because I require this
451  // data
452  auto dimensions = get_dimensions(spec);
453  if (!dimensions.status().ok()) {
454  return dimensions.status();
455  }
456 
457  std::unordered_map<std::string, uint64_t> dimensionMap = dimensions.value();
458 
459  std::vector<nlohmann::json> datasetSpec;
460  for (auto& variable : spec["variables"]) {
461  auto variableSpec = from_json_to_spec(variable, dimensionMap, path);
462  if (!variableSpec.status().ok()) {
463  return variableSpec.status();
464  }
465  datasetSpec.emplace_back(variableSpec.value());
466  }
467  if (!spec.contains("metadata")) {
468  spec["metadata"] = nlohmann::json::object();
469  }
470  return std::make_tuple(spec["metadata"], datasetSpec);
471 }
