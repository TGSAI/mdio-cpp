55 Result<nlohmann::json> get_zarray(const ::nlohmann::json metadata) {
56  // derive .zarray json metadata (without reading it).
57  auto json =
58  metadata; // Why am I doing this? It's an extra copy that does nothing!
59  nlohmann::json zarray;
60  if (!json.contains("metadata")) {
61  json["metadata"] = nlohmann::json::object(); // Just add an empty object
62  json["metadata"]["attributes"] =
63  nlohmann::json::object(); // We need attributes as well
64  }
65 
66  // these fields can have defaults:
67  if (!json["metadata"].contains("order")) {
68  zarray["order"] = "C";
69  } else {
70  zarray["order"] = json["metadata"]["order"];
71  }
72  if (!json["metadata"].contains("filters")) {
73  zarray["filters"] = nullptr;
74  } else {
75  zarray["filters"] = json["metadata"]["filters"];
76  }
77 
78  if (!json["metadata"].contains("fill_value")) {
79  zarray["fill_value"] = nullptr;
80  } else {
81  zarray["fill_value"] = json["metadata"]["fill_value"];
82  }
83 
84  if (!json["metadata"].contains("zarr_format")) {
85  zarray["zarr_format"] = 2;
86  } else {
87  zarray["zarr_format"] = json["metadata"]["zarr_format"];
88  }
89 
90  if (!json["metadata"].contains("chunks") &&
91  json["metadata"].contains("shape")) {
92  zarray["chunks"] = json["metadata"]["shape"];
93  } else {
94  zarray["chunks"] = json["metadata"]["chunks"];
95  }
96 
97  if (!json["metadata"].contains("compressor")) {
98  zarray["compressor"] = nullptr;
99  } else {
100  zarray["compressor"] = json["metadata"]["compressor"];
101  }
102 
103  if (!json["metadata"].contains("dimension_separator")) {
104  zarray["dimension_separator"] = "/";
105  } else {
106  zarray["dimension_separator"] = json["metadata"]["dimension_separator"];
107  }
108 
109  zarray["shape"] = json["metadata"]["shape"];
110  zarray["dtype"] = json["metadata"]["dtype"];
111 
112  // fixme chunks must be configured ...
113  MDIO_ASSIGN_OR_RETURN(
114  auto zarr_metadata,
115  tensorstore::internal_zarr::ZarrMetadata::FromJson(zarray))
116 
117  return ::nlohmann::json(zarr_metadata);
118 }

127 Future<void> write_zmetadata(
128  const ::nlohmann::json& dataset_metadata,
129  const std::vector<::nlohmann::json>& json_variables) {
130  // header material at the root of the dataset ...
131  // Configure a kvstore (we can't deduce if it's in memory etc).
132  // {
133  // "kvstore",
134  // {
135  // {"driver", "file"},
136  // {"path", "name"}
137  // }
138  //}
139  auto zattrs = dataset_metadata;
140 
141  // FIXME - generalize for zarr v3
142  ::nlohmann::json zgroup;
143  zgroup["zarr_format"] = 2;
144 
145  // The consolidated metadata for the datset
146  ::nlohmann::json zmetadata;
147 
148  // FIXME - don't hard code here ...
149  zmetadata["zarr_consolidated_format"] = 1;
150 
151  zmetadata["metadata"][".zattrs"] = zattrs;
152  zmetadata["metadata"][".zgroup"] = zgroup;
153 
154  std::string zarray_key;
155  std::string zattrs_key;
156  std::string driver =
157  json_variables[0]["kvstore"]["driver"].get<std::string>();
158 
159  for (const auto& json : json_variables) {
160  zarray_key =
161  std::filesystem::path(json["kvstore"]["path"]).stem() / ".zarray";
162  zattrs_key =
163  std::filesystem::path(json["kvstore"]["path"]).stem() / ".zattrs";
164 
165  MDIO_ASSIGN_OR_RETURN(zmetadata["metadata"][zarray_key], get_zarray(json))
166 
167  nlohmann::json fixedJson = json["attributes"];
168  fixedJson["_ARRAY_DIMENSIONS"] = fixedJson["dimension_names"];
169  fixedJson.erase("dimension_names");
170  // We do not want to be seralizing the variable_name. It should be
171  // self-describing
172  if (fixedJson.contains("variable_name")) {
173  fixedJson.erase("variable_name");
174  }
175  if (fixedJson.contains("long_name") &&
176  fixedJson["long_name"].get<std::string>() == "") {
177  fixedJson.erase("long_name");
178  }
179  if (fixedJson.contains("metadata")) {
180  if (fixedJson["metadata"].contains("chunkGrid")) {
181  fixedJson["metadata"].erase("chunkGrid");
182  }
183  for (auto& item : fixedJson["metadata"].items()) {
184  fixedJson[item.key()] = std::move(item.value());
185  }
186  fixedJson.erase("metadata");
187  }
188  // Case where an empty array of coordinates were provided
189  if (fixedJson.contains("coordinates")) {
190  auto coords = fixedJson["coordinates"];
191  if (coords.empty() ||
192  (coords.is_string() && coords.get<std::string>() == "")) {
193  fixedJson.erase("coordinates");
194  }
195  }
196  zmetadata["metadata"][zattrs_key] = fixedJson;
197  }
198 
199  nlohmann::json kvstore = nlohmann::json::object();
200  kvstore["driver"] = driver;
201  std::vector<std::string> file_parts = absl::StrSplit(
202  json_variables[0]["kvstore"]["path"].get<std::string>(), '/');
203  size_t toRemove = file_parts.back().size();
204  std::string strippedPath =
205  json_variables[0]["kvstore"]["path"].get<std::string>().substr(
206  0, json_variables[0]["kvstore"]["path"].get<std::string>().size() -
207  toRemove - 1);
208  kvstore["path"] = strippedPath;
209 
210  if (driver == "gcs" || driver == "s3") {
211  kvstore["bucket"] =
212  json_variables[0]["kvstore"]["bucket"].get<std::string>();
213  std::string cloudPath = kvstore["path"].get<std::string>();
214  kvstore["path"] = cloudPath;
215  }
216 
217  auto kvs_future = tensorstore::kvstore::Open(kvstore);
218 
219  auto zattrs_future = tensorstore::MapFutureValue(
220  tensorstore::InlineExecutor{},
221  [zattrs = std::move(zattrs)](const tensorstore::KvStore& kvstore) {
222  return tensorstore::kvstore::Write(kvstore, "/.zattrs",
223  absl::Cord(zattrs.dump(4)));
224  },
225  kvs_future);
226 
227  auto zmetadata_future = tensorstore::MapFutureValue(
228  tensorstore::InlineExecutor{},
229  [zmetadata = std::move(zmetadata)](const tensorstore::KvStore& kvstore) {
230  return tensorstore::kvstore::Write(kvstore, "/.zmetadata",
231  absl::Cord(zmetadata.dump(4)));
232  },
233  kvs_future);
234 
235  auto zgroup_future = tensorstore::MapFutureValue(
236  tensorstore::InlineExecutor{},
237  [zgroup = std::move(zgroup)](const tensorstore::KvStore& kvstore) {
238  return tensorstore::kvstore::Write(kvstore, "/.zgroup",
239  absl::Cord(zgroup.dump(4)));
240  },
241  kvs_future);
242 
243  return tensorstore::WaitAllFuture(zattrs_future, zmetadata_future,
244  zgroup_future);
245 }

254 Future<tensorstore::KvStore> dataset_kvs_store(
255  const std::string& dataset_path) {
256  // the tensorstore driver needs a bucket field
257  ::nlohmann::json kvstore;
258 
259  absl::string_view output_file = dataset_path;
260 
261  if (absl::StartsWith(output_file, "gs://")) {
262  absl::ConsumePrefix(&output_file, "gs://");
263  kvstore["driver"] = "gcs";
264  } else if (absl::StartsWith(output_file, "s3://")) {
265  absl::ConsumePrefix(&output_file, "s3://");
266  kvstore["driver"] = "s3";
267  } else {
268  kvstore["driver"] = "file";
269  kvstore["path"] = output_file;
270  return tensorstore::kvstore::Open(kvstore);
271  } // FIXME - we need azure support ...
272 
273  std::vector<std::string> file_parts = absl::StrSplit(output_file, '/');
274  if (file_parts.size() < 2) {
275  return absl::InvalidArgumentError(
276  "gcs/s3 drivers requires [s3/gs]://[bucket]/[path_to_file]");
277  }
278 
279  std::string bucket = file_parts[0];
280  std::string filepath(file_parts[1]);
281  for (std::size_t i = 2; i < file_parts.size(); ++i) {
282  filepath += "/" + file_parts[i];
283  }
284  // update the bucket and path ...
285  kvstore["bucket"] = bucket;
286  kvstore["path"] = filepath;
287 
288  return tensorstore::kvstore::Open(kvstore);
289 }

300 Future<std::tuple<::nlohmann::json, std::vector<::nlohmann::json>>>
301 from_zmetadata(const std::string& dataset_path) {
302  // e.g. dataset_path = "zarrs/acceptance/";
303  // FIXME - enable async
304  auto kvs_future = mdio::internal::dataset_kvs_store(dataset_path).result();
305 
306  if (!kvs_future.ok()) {
307  return internal::CheckMissingDriverStatus(kvs_future.status());
308  }
309  auto kvs_read_result =
310  tensorstore::kvstore::Read(kvs_future.value(), ".zmetadata").result();
311  if (!kvs_read_result.ok()) {
312  return internal::CheckMissingDriverStatus(kvs_read_result.status());
313  }
314 
315  ::nlohmann::json zmetadata;
316  try {
317  zmetadata =
318  ::nlohmann::json::parse(std::string(kvs_read_result.value().value));
319  } catch (const nlohmann::json::parse_error& e) {
320  // It's a common error to not have a trailing slash on the dataset path.
321  if (!dataset_path.empty() && dataset_path.back() != '/') {
322  std::string fixPath = dataset_path + "/";
323  return mdio::internal::from_zmetadata(fixPath);
324  }
325  return absl::Status(absl::StatusCode::kInvalidArgument, e.what());
326  }
327 
328  if (!zmetadata.contains("metadata")) {
329  return absl::Status(absl::StatusCode::kInvalidArgument,
330  "zmetadata does not contain metadata.");
331  }
332 
333  if (!zmetadata["metadata"].contains(".zattrs")) {
334  return absl::Status(absl::StatusCode::kInvalidArgument,
335  "zmetadata does not contain dataset metadata.");
336  }
337 
338  auto dataset_metadata = zmetadata["metadata"][".zattrs"];
339 
340  std::string driver = "file";
341  // TODO(BrianMichell): Make this more robust. May be invalid if the stored
342  // path gets mangled somehow. Infer the driver
343  if (dataset_path.length() > 5) {
344  if (dataset_path.substr(0, 5) == "gs://") {
345  driver = "gcs";
346  } else if (dataset_path.substr(0, 5) == "s3://") {
347  driver = "s3";
348  }
349  }
350 
351  std::string bucket;
352  std::string cloudPath;
353  if (driver != "file") {
354  std::string providedPath =
355  dataset_path.substr(5); // Strip the gs:// or s3://
356  size_t bucketLen = providedPath.find_first_of('/');
357  bucket = providedPath.substr(0, bucketLen); // Extract the bucket name
358  cloudPath = providedPath.substr(
359  bucketLen + 1, providedPath.length() - 2); // Extract the path
360  }
361 
362  // Remove .zattrs from metadata
363  zmetadata["metadata"].erase(".zattrs");
364  std::vector<nlohmann::json> json_vars_from_zmeta;
365  // Assemble a list of json for opening the variables in the dataset.
366  for (auto& element : zmetadata["metadata"].items()) {
367  // FIXME - remove hard code .zarray
368  if (element.key().substr(element.key().find_last_of(".") + 1) == "zarray") {
369  std::string variable_name =
370  element.key().substr(0, element.key().find("/"));
371  nlohmann::json new_dict = {
372  {"driver", "zarr"},
373  {"kvstore",
374  {{"driver", driver}, {"path", dataset_path + "/" + variable_name}}}};
375  if (driver != "file") {
376  new_dict["kvstore"]["bucket"] = bucket;
377  new_dict["kvstore"]["path"] = cloudPath + variable_name;
378  }
379  json_vars_from_zmeta.push_back(new_dict);
380  }
381  }
382  if (!json_vars_from_zmeta.size()) {
383  return absl::Status(absl::StatusCode::kInvalidArgument,
384  "Not variables found in zmetadata.");
385  }
386 
387  return tensorstore::ReadyFuture<
388  std::tuple<::nlohmann::json, std::vector<::nlohmann::json>>>(
389  std::make_tuple(dataset_metadata, json_vars_from_zmeta));
390 }

460  template <typename... DimensionIdentifier>
461  mdio::Result<std::vector<Variable<>::Interval>> get_intervals(
462  const DimensionIdentifier&... labels) const {
463  std::vector<Variable<>::Interval> intervals;
464  std::unordered_set<std::string_view> labels_set;
465  auto idents = variables.get_iterable_accessor();
466  for (auto& ident : idents) {
467  MDIO_ASSIGN_OR_RETURN(auto var, variables.at(ident));
468  auto intervalRes = var.get_intervals(labels...);
469  if (intervalRes.status().ok()) {
470  for (auto& interval : intervalRes.value()) {
471  if (labels_set.count(interval.label.label()) == 0) {
472  labels_set.insert(interval.label.label());
473  intervals.push_back(interval);
474  }
475  }
476  }
477  }
478 
479  if (intervals.empty()) {
480  return absl::NotFoundError("No intervals found for the given labels.");
481  }
482  return intervals;
483  }

533  template <typename... Descriptors>
534  Result<Dataset> isel(Descriptors&... descriptors) {
535  VariableCollection vars;
536 
537  // the shape of the new domain
538  std::map<std::string, tensorstore::IndexDomainDimension<>> dims;
539  std::vector<std::string> keys = variables.get_iterable_accessor();
540 
541  for (const auto& name : keys) {
542  MDIO_ASSIGN_OR_RETURN(auto variable,
543  variables.at(name).value().slice(
544  std::forward<Descriptors>(descriptors)...))
545  // add to variable
546  vars.add(name, variable);
547 
548  // FIXME - check consistent dims ...
549  DimensionIndex idx = 0;
550  for (const auto label : variable.get_store().domain().labels()) {
551  // structarrays must have a byte dimension.
552  if (!label.empty()) {
553  dims[label] = variable.get_store().domain()[idx];
554  }
555  ++idx;
556  }
557  }
558 
559  size_t size = dims.size();
560  std::vector<std::string> labels(size);
561  std::vector<Index> origin(size);
562  std::vector<Index> shape(size);
563 
564  DimensionIndex idx = 0;
565  for (const auto& [key, val] : dims) {
566  labels[idx] = key;
567  origin[idx] = val.interval().inclusive_min();
568  shape[idx] = val.interval().size();
569  ++idx;
570  }
571 
572  MDIO_ASSIGN_OR_RETURN(auto new_domain,
573  tensorstore::IndexDomainBuilder<>(size)
574  .origin(origin)
575  .shape(shape)
576  .labels(labels)
577  .Finalize())
578  return Dataset{metadata, vars, coordinates, new_domain};
579  }

612  // Wrapper function that generates the index sequence
621  Result<Dataset> isel(const std::vector<RangeDescriptor<Index>>& slices) {
622  if (slices.empty()) {
623  return absl::InvalidArgumentError("No slices provided.");
624  }
625 
626  if (slices.size() > internal::kMaxNumSlices) {
627  return absl::InvalidArgumentError(
628  absl::StrCat("Too many slices provided or implicitly generated. "
629  "Maximum number of slices is ",
630  internal::kMaxNumSlices, " but ", slices.size(),
631  " were provided.\n\tUse -DMAX_NUM_SLICES cmake flag to "
632  "increase the maximum number of slices."));
633  }
634 
635  std::vector<RangeDescriptor<Index>> slicesCopy = slices;
636  for (int i = slices.size(); i <= internal::kMaxNumSlices; i++) {
637  slicesCopy.emplace_back(
638  RangeDescriptor<Index>({internal::kInertSliceKey, 0, 1, 1}));
639  }
640 
641  // Generate the index sequence and call the implementation
642  return call_isel_with_vector_impl(
643  slicesCopy, std::make_index_sequence<internal::kMaxNumSlices>{});
644  }

650  template <typename... Descriptors>
651  Result<std::map<std::string_view, std::vector<Index>>> descriptor_to_index(
652  Descriptors... descriptors) {
653  absl::Status trueStatus =
654  absl::OkStatus(); // A hack to allow for true error status return.
655  std::map<std::string_view, std::vector<Index>> label_to_indices;
656  auto processDescriptor = [this, &label_to_indices,
657  &trueStatus](auto& descriptor) -> absl::Status {
658  using ValueType =
659  typename extract_descriptor_Ttype<decltype(descriptor)>::type;
660 
661  auto varRes =
662  variables.get<ValueType>(std::string(descriptor.label.label()));
663  if (!varRes.status().ok()) {
664  trueStatus = varRes.status();
665  return trueStatus;
666  }
667  auto var = varRes.value();
668  auto varFut = var.Read();
669  if (!varFut.status().ok()) {
670  trueStatus = varFut.status();
671  return trueStatus;
672  }
673  auto varDat = varFut.value();
674  auto varAccessor = varDat.get_data_accessor();
675 
676  std::vector<Index> indices;
677  auto offset = varDat.get_flattened_offset();
678  if constexpr ((std::is_same_v<
679  Descriptors,
680  ListDescriptor<typename Descriptors::type>> &&
681  ...)) {
682  std::set<ValueType> values;
683  for (auto val : descriptor.values) {
684  if (values.count(val) > 0) {
685  trueStatus = absl::InvalidArgumentError(
686  "Repeated value found in ListDescriptor.");
687  return trueStatus;
688  }
689  values.insert(val);
690  bool found = false;
691  for (Index i = offset; i < var.num_samples() + offset; ++i) {
692  if (varAccessor({i}) == val) {
693  if (found) {
694  trueStatus = absl::InvalidArgumentError(
695  "Repeated value found in ListDescriptor.");
696  return trueStatus;
697  }
698  label_to_indices[descriptor.label.label()].push_back(i);
699  found = true;
700  }
701  }
702  if (!found) {
703  trueStatus = absl::InvalidArgumentError(
704  "Value not found in ListDescriptor.");
705  return trueStatus;
706  }
707  }
708  } else {
709  // We must check for every occurance of the value
710  for (Index i = offset; i < var.num_samples() + offset; ++i) {
711  if (varAccessor({i}) == descriptor.value) {
712  label_to_indices[descriptor.label.label()].push_back(i);
713  }
714  }
715  }
716 
717  // Add the indices to the map
718  return absl::OkStatus();
719  };
720 
721  auto status = (processDescriptor(descriptors).ok() && ...);
722  if (!status) {
723  return trueStatus;
724  }
725 
726  return label_to_indices;
727  }

737  template <typename... Descriptors>
738  Result<Dataset> sel(Descriptors... descriptors) {
739  /*
740  Case 1: ValueDescriptor with repeated values: Get all occurrences of the
741  value Case 2: ListDescriptor with repeated values (single element): Return
742  Invalid Reindexing error Case 3: ListDescriptor with repeated values
743  (multiple elements): Return Invalid Reindexing error (Case 2) Case 4: Same
744  as Case 1 Case 5: RangeDescriptor with repeated values: Return Invalid
745  Reindexing error Case 6: RangeDescriptor with unique values: Get the range
746  from start to stop, include everything in-between
747 
748  Case fail: label is not 1D
749  Case fail: label is repeated. This is a dictionary in xarray, so not
750  allowed.
751  */
752 
753  /*
754  Valid cases:
755  Case 1: ValueDescriptor with unique value: Get the single value.
756  No error state.
757  Get the index and convert to conventional isel.
758  Case 2: ValueDescriptor with non-unique value: Get all occurrences of the
759  value. No error state. Case 3: ListDescriptor with unique values: Get the
760  individual values. Error state for repeated values. Case 4: RangeDescriptor
761  with unique start and stop values: Get the range from start to stop, include
762  everything in-between. Error state for repeated start/stop values. Any
763  repeated values that are not start/stop are fair game. Get the start and
764  stop indicies and create a single isel.
765  */
766 
767  // Check that all descriptors are of the same outer type
768  if (!are_same<Descriptors...>()) {
769  return absl::InvalidArgumentError(
770  "All descriptors must be of the same type.");
771  }
772 
773  // Validate each descriptor (for example, ListDescriptor not yet supported)
774  auto validateDescriptors = [this](auto& descriptor) {
775  using DescriptorType = typename outer_type<decltype(descriptor)>::type;
776  if constexpr (std::is_same_v<
777  std::remove_reference_t<DescriptorType>,
778  ListDescriptor<typename std::remove_reference_t<
779  decltype(descriptor)>::type>>) {
780  return absl::UnimplementedError(
781  "Support for ListDescriptor is not yet implemented.");
782  }
783  // TODO(BrianMichell): Remove this check when SliceDescriptor is removed
784  if constexpr (std::is_same_v<DescriptorType, SliceDescriptor>) {
785  return absl::InvalidArgumentError(
786  "SliceDescriptor is deprecated and will be removed in future "
787  "versions. Please use RangeDescriptor instead.\nThe sel method "
788  "does not support SliceDescriptor.");
789  }
790 
791  MDIO_ASSIGN_OR_RETURN(
792  auto var, variables.at(std::string(descriptor.label.label())));
793  if (var.dimensions().rank() != 1) {
794  return absl::InvalidArgumentError("Label must be 1D.");
795  }
796 
797  return absl::OkStatus();
798  };
799 
800  std::set<std::string_view> labels;
801 
802  // Call validateDescriptors for each descriptor
803  {
804  // Manage the scope of status
805  absl::Status status;
806  for (auto& descriptor : {descriptors...}) {
807  status = validateDescriptors(descriptor);
808  if (!status.ok()) {
809  return status;
810  }
811  if (labels.count(descriptor.label.label()) > 0) {
812  return absl::InvalidArgumentError("Label must not be repeated.");
813  }
814  if (descriptor.label.index() !=
815  std::numeric_limits<DimensionIndex>::max()) {
816  return absl::InvalidArgumentError(
817  "Expected label to be a dimension name but got an index.");
818  }
819  labels.insert(descriptor.label.label());
820  }
821  }
822 
823  // Check if the descriptors are of type ValueDescriptor
824  if constexpr ((std::is_same_v<
825  Descriptors,
826  ValueDescriptor<typename Descriptors::type>> &&
827  ...)) {
828  auto slicer = descriptor_to_index(descriptors...);
829  if (!slicer.status().ok()) {
830  return slicer.status();
831  }
832 
833  auto label_to_indices = slicer.value();
834 
835  // Build out the slice descriptors
836  std::vector<RangeDescriptor<Index>> slices;
837  for (auto& elem : label_to_indices) {
838  auto size = elem.second.size();
839  for (int i = 0; i < size; ++i) {
840  slices.emplace_back(RangeDescriptor<Index>(
841  {elem.first, elem.second[i], elem.second[i] + 1, 1}));
842  }
843  }
844 
845  if (slices.empty()) {
846  return absl::InvalidArgumentError(
847  "No slices could be made from the given descriptors.");
848  }
849  // The map 'label_to_indices' is now populated with all the relevant
850  // indices. You can now proceed with further processing based on this map.
851 
852  return isel(slices);
853  } else if constexpr ((std::is_same_v</*NOLINT: readability/braces*/
854  Descriptors,
855  ListDescriptor<
856  typename Descriptors::type>> &&
857  ...)) {
858  auto slicer = descriptor_to_index(descriptors...);
859  if (!slicer.status().ok()) {
860  return slicer.status();
861  }
862 
863  auto label_to_indices = slicer.value();
864 
865  // Build out the slice descriptors
866  std::vector<RangeDescriptor<Index>> slices;
867  for (auto& elem : label_to_indices) {
868  auto size = elem.second.size();
869  for (int i = 0; i < size; ++i) {
870  slices.emplace_back(RangeDescriptor<Index>(
871  {elem.first, elem.second[i], elem.second[i] + 1, 1}));
872  }
873  }
874 
875  if (slices.empty()) {
876  return absl::InvalidArgumentError(
877  "No slices could be made from the given descriptors.");
878  }
879  // The map 'label_to_indices' is now populated with all the relevant
880  // indices. You can now proceed with further processing based on this map.
881 
882  return isel(slices);
883  } else {
884  std::map<std::string_view, std::pair<Index, Index>>
885  label_to_range; // pair.first = start, pair.second = stop
886  absl::Status trueStatus =
887  absl::OkStatus(); // A hack to allow for true error status return.
888 
889  auto processDescriptor = [this, &label_to_range,
890  &trueStatus](auto& descriptor) -> absl::Status {
891  using ValueType =
892  typename extract_descriptor_Ttype<decltype(descriptor)>::type;
893 
894  if (descriptor.start == descriptor.stop) {
895  trueStatus = absl::InvalidArgumentError(
896  "Start and stop values must be different.");
897  return trueStatus;
898  }
899 
900  auto varRes =
901  variables.get<ValueType>(std::string(descriptor.label.label()));
902  if (!varRes.status().ok()) {
903  trueStatus = varRes.status();
904  return trueStatus;
905  }
906  auto var = varRes.value();
907  auto varFut = var.Read();
908  if (!varFut.status().ok()) {
909  trueStatus = varFut.status();
910  return trueStatus;
911  }
912  auto varDat = varFut.value();
913  auto varAccessor = varDat.get_data_accessor();
914 
915  std::pair<bool, Index> start = {false, 0};
916  std::pair<bool, Index> stop = {false, 0};
917  auto offset = varDat.get_flattened_offset();
918 
919  for (Index i = offset; i < var.num_samples() + offset; i++) {
920  if (varAccessor({i}) == descriptor.start) {
921  if (start.first) {
922  trueStatus = absl::InvalidArgumentError("Repeated start value.");
923  return trueStatus;
924  }
925  start = {true, i};
926  }
927  if (varAccessor({i}) == descriptor.stop) {
928  if (stop.first) {
929  trueStatus = absl::InvalidArgumentError("Repeated stop value.");
930  return trueStatus;
931  }
932  stop = {true, i};
933  }
934  }
935 
936  if (!start.first) {
937  trueStatus = absl::InvalidArgumentError("Start value not found.");
938  return trueStatus;
939  }
940  if (!stop.first) {
941  trueStatus = absl::InvalidArgumentError("Stop value not found.");
942  return trueStatus;
943  }
944 
945  // Xarray behavior is to effectively remove the Variable in this case.
946  if (start.second >= stop.second) {
947  trueStatus = absl::UnimplementedError(
948  "Start value happens after stop value. This is not a supported "
949  "case.");
950  return trueStatus;
951  }
952  // This case should be caught by the earlier check, but it's here for
953  // completeness.
954  if (label_to_range.count(descriptor.label.label()) > 0) {
955  trueStatus =
956  absl::InvalidArgumentError("Label must not be repeated.");
957  return trueStatus;
958  }
959  label_to_range[descriptor.label.label()] = {start.second, stop.second};
960  return absl::OkStatus();
961  };
962 
963  auto status = (processDescriptor(descriptors).ok() && ...);
964  if (!status) {
965  return trueStatus;
966  }
967 
968  std::vector<RangeDescriptor<Index>> slices;
969  for (auto& elem : label_to_range) {
970  slices.emplace_back(RangeDescriptor<Index>(
971  {elem.first, elem.second.first, elem.second.second + 1, 1}));
972  }
973 
974  if (slices.empty()) {
975  return absl::InvalidArgumentError(
976  "No slices could be made from the given descriptors.");
977  }
978 
979  return isel(slices);
980  }
981 
982  return absl::OkStatus();
983  }
