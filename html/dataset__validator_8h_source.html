45 absl::Status validate_schema(nlohmann::json& spec /*NOLINT*/) {
46  nlohmann::json targetSchema =
47  nlohmann::json::parse(kDatasetSchema, nullptr, false);
48  if (targetSchema.is_discarded()) {
49  return absl::NotFoundError("Failed to load schema");
50  }
51 
52  nlohmann::json_schema::json_validator validator(
53  nullptr, nlohmann::json_schema::default_string_format_check);
54  validator.set_root_schema(targetSchema);
55 
56  try {
57  validator.validate(spec);
58  } catch (const std::exception& e) {
59  return absl::Status(
60  absl::StatusCode::kInvalidArgument,
61  "Validation failed, here is why: " + std::string(e.what()));
62  }
63 
64  return absl::OkStatus();
65 }

75 absl::Status validate_coordinates_present(const nlohmann::json& spec) {
76  // Build a mapping of all the dimension coordinates
77  std::unordered_set<std::string>
78  dimension; // name of all 1-d Variables who's name matches the dimension
79  // name
80  // Using this will ensure that a variable["coordinates"][i] has a defined
81  // variable["name"]
82  std::unordered_set<std::string> variables; // Name of all Variables
83  for (auto& variable : spec["variables"]) {
84  variables.insert(variable["name"].dump());
85  if (variable["dimensions"][0].is_object()) {
86  if (variable["dimensions"].size() == 1 &&
87  variable["dimensions"][0]["name"] == variable["name"]) {
88  // This must be a dimension coordinate -- Must exist if a Variable has
89  // the dimension
90  dimension.insert(variable["name"].dump());
91  }
92  }
93  }
94 
95  // Make sure all the Variables have a dimension coordinate
96  for (auto& variable : spec["variables"]) {
97  // No need to verify a dimension coordinate
98  if (contains(dimension, variable["name"].dump())) {
99  continue;
100  }
101 
102  if (variable["dimensions"][0].is_object()) { // list[NamedDimensions]
103  for (auto& dimensionName : variable["dimensions"]) {
104  if (!contains(dimension, dimensionName["name"].dump())) {
105  return absl::Status(absl::StatusCode::kInvalidArgument,
106  "Variable " + variable["name"].dump() +
107  " has a dimension " +
108  dimensionName["name"].dump() +
109  " that is not a dimension coordinate.");
110  }
111  }
112  } else { // list[str]
113  for (auto& dimensionName : variable["dimensions"]) {
114  if (!contains(dimension, dimensionName.dump())) {
115  return absl::Status(absl::StatusCode::kInvalidArgument,
116  "Variable " + variable["name"].dump() +
117  " has a dimension " + dimensionName.dump() +
118  " that is not a dimension coordinate.");
119  }
120  }
121  }
122 
123  // TODO(BrianMichell): Implement support for list[Coordinate] later on
124  if (variable.contains("coordinates")) {
125  for (auto& coordinate : variable["coordinates"]) {
126  if (!contains(variables, coordinate.dump())) {
127  return absl::Status(absl::StatusCode::kInvalidArgument,
128  "Variable " + variable["name"].dump() +
129  " has a coordinate " + coordinate.dump() +
130  " that is not a Variable.");
131  }
132  }
133  }
134  }
135 
136  return absl::OkStatus();
137 }

148 absl::Status validate_dataset(nlohmann::json& spec /*NOLINT*/) {
149  absl::Status schemaStatus = validate_schema(spec);
150  if (!schemaStatus.ok()) {
151  return schemaStatus;
152  }
153 
154  absl::Status coordinateStatus = validate_coordinates_present(spec);
155  if (!coordinateStatus.ok()) {
156  return coordinateStatus;
157  }
158 
159  return absl::OkStatus();
160 }
