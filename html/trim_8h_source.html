44 template <typename... Descriptors>
45 Future<void> TrimDataset(std::string dataset_path,
46  bool delete_sliced_out_chunks,
47  const Descriptors&... descriptors) {
48  // Open the dataset
49  auto dsRes = mdio::Dataset::Open(dataset_path, mdio::constants::kOpen);
50  if (!dsRes.status().ok()) {
51  return dsRes.status();
52  }
53  mdio::Dataset ds = dsRes.value();
54  // Trim the dataset
55  std::unordered_map<std::string_view, mdio::Index> shapeDescriptors;
56  std::vector<mdio::RangeDescriptor<mdio::Index>> descriptorList = {
57  descriptors...};
58  if (descriptorList.size() == 0) {
59  // No slices = no op
60  return absl::OkStatus();
61  }
62  for (const auto& descriptor : descriptorList) {
63  shapeDescriptors[descriptor.label.label()] = descriptor.stop;
64  }
65 
66  for (auto& varIdentifier : ds.variables.get_iterable_accessor()) {
67  MDIO_ASSIGN_OR_RETURN(auto var, ds.variables.at(varIdentifier))
68  var.set_metadata_publish_flag(true);
69 
70  bool wasStruct = var.dimensions().labels().back() == "";
71 
72  auto varStore = var.get_store();
73  std::vector<tensorstore::Index> implicitDims;
74  std::vector<tensorstore::Index> newShape;
75 
76  auto dims = var.dimensions().shape().size();
77  if (wasStruct) {
78  --dims;
79  }
80 
81  for (size_t i = 0; i < dims; i++) {
82  implicitDims.push_back(tensorstore::kImplicit);
83  if (shapeDescriptors.count(var.dimensions().labels()[i]) > 0) {
84  newShape.push_back(shapeDescriptors[var.dimensions().labels()[i]]);
85  } else {
86  newShape.push_back(var.dimensions().shape()[i]);
87  }
88  }
89 
90  if (wasStruct) {
91  implicitDims.push_back(tensorstore::kImplicit);
92  newShape.push_back(tensorstore::kImplicit);
93  }
94 
95  tensorstore::ResizeOptions resizeOptions;
96  if (delete_sliced_out_chunks) {
97  resizeOptions.mode = tensorstore::ResizeMode::resize_tied_bounds;
98  } else if (wasStruct) {
99  resizeOptions.mode = tensorstore::ResizeMode::resize_metadata_only;
100  } else {
101  resizeOptions.mode = tensorstore::ResizeMode::resize_metadata_only;
102  }
103 
104  auto resizedStatus = tensorstore::Resize(
105  varStore, tensorstore::span<const tensorstore::Index>(implicitDims),
106  tensorstore::span<const tensorstore::Index>(newShape), resizeOptions);
107 
108  if (!resizedStatus.status().ok()) {
109  return resizedStatus.status();
110  }
111  }
112 
113  return ds.CommitMetadata();
114 }
