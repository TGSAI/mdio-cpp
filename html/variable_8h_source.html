207 absl::Status CheckMissingDriverStatus(const absl::Status& status) {
208  std::string error(status.message());
209  if (error.find("Error parsing object member \"driver\"") !=
210  std::string::npos) {
211  if (error.find("is not registered") != std::string::npos) {
212  if (error.find("gcs") != std::string::npos) {
213  return absl::InvalidArgumentError(
214  "A GCS path was detected but the GCS driver was not "
215  "registered.\nPlease ensure that your CMake includes the "
216  "mdio_INTERNAL_GCS_DRIVER_DEPS variable.");
217  } else if (error.find("s3") != std::string::npos) {
218  return absl::InvalidArgumentError(
219  "An S3 path was detected but the S3 driver was not "
220  "registered.\nPlease ensure that your CMake includes the "
221  "mdio_INTERNAL_S3_DRIVER_DEPS variable.");
222  } else {
223  return absl::InvalidArgumentError(
224  "An unexpected driver registration error has occured. Please file "
225  "a bug report with the error message to "
226  "https://github.com/TGSAI/mdio-cpp/issues\n" +
227  error);
228  }
229  }
230  }
231  return status;
232 }

250 template <ReadWriteMode Mode = ReadWriteMode::dynamic>
251 Result<std::tuple<nlohmann::json, nlohmann::json>> ValidateAndProcessJson(
252  const nlohmann::json& json_spec) {
253  // Check if attributes are in the original JSON
254  if (!json_spec.contains("attributes")) {
255  return absl::InvalidArgumentError(
256  "The json_spec does not contain 'attributes'.");
257  }
258 
259  // Check if dimensions field exists in attributes
260  if (!json_spec["attributes"].contains("dimension_names")) {
261  return absl::InvalidArgumentError(
262  "The 'attributes' does not contain 'dimension_names'.");
263  }
264 
265  nlohmann::json json_for_store = json_spec;
266 
267  // remove attributes, the v2 store won't parse it.
268  json_for_store.erase("attributes");
269 
270  // Create a new JSON and add the kvstore
271  nlohmann::json new_json = json_spec["attributes"];
272  // update the variable name
273  new_json["variable_name"] =
274  std::filesystem::path(json_spec["kvstore"]["path"]).stem().string();
275 
276  return std::make_tuple(json_for_store, new_json);
277 }

294 template <typename T = void, DimensionIndex R = dynamic_rank,
295  ReadWriteMode M = ReadWriteMode::dynamic>
296 Result<Variable<T, R, M>> from_json(
297  const ::nlohmann::json spec,
298  const tensorstore::TensorStore<T, R, M> store) {
299  auto attributes = spec;
300  // There seems to be a case where we don't actually strip out the super
301  // attributes field by this point.
302  if (attributes.contains("attributes") &&
303  attributes["attributes"].contains("variable_name")) {
304  attributes = attributes["attributes"];
305  }
306 
307  if (!attributes.contains("variable_name")) {
308  return absl::NotFoundError("Could not find Variable's name.");
309  }
310  std::string name = attributes["variable_name"];
311 
312  nlohmann::json scrubbed_spec = attributes;
313  scrubbed_spec.erase("variable_name");
314  auto specWithoutChunkgrid = spec;
315  if (specWithoutChunkgrid.contains("metadata")) {
316  if (specWithoutChunkgrid["metadata"].contains("chunkGrid")) {
317  specWithoutChunkgrid["metadata"].erase("chunkGrid");
318  }
319  }
320  auto attrsRes = UserAttributes::FromVariableJson(specWithoutChunkgrid);
321  if (!attrsRes.ok()) {
322  return attrsRes.status();
323  }
324 
325  std::string long_name =
326  ""; // Default case. TODO: Look into making this optional.
327  if (scrubbed_spec.contains("long_name")) {
328  if (scrubbed_spec["long_name"].is_string() &&
329  scrubbed_spec["long_name"].get<std::string>().size() > 0) {
330  long_name = scrubbed_spec["long_name"].get<std::string>();
331  } else {
332  scrubbed_spec.erase("long_name");
333  }
334  } // Not having this field is fine. If it doesn't get seralized returning an
335  // error status is not necessary.
336 
337  // These live in our `UserAttributes` object and are technically mutable. The
338  // best kind of mutable!
339  if (scrubbed_spec.contains("metadata")) {
340  if (scrubbed_spec["metadata"].contains("attributes")) {
341  scrubbed_spec["metadata"].erase("attributes");
342  }
343  if (scrubbed_spec["metadata"].contains("statsV1")) {
344  scrubbed_spec["metadata"].erase("statsV1");
345  }
346  if (scrubbed_spec["metadata"].contains("unitsV1")) {
347  scrubbed_spec["metadata"].erase("unitsV1");
348  }
349  }
350 
351  std::shared_ptr<std::shared_ptr<UserAttributes>> attrs =
352  std::make_shared<std::shared_ptr<UserAttributes>>(
353  std::make_shared<UserAttributes>(attrsRes.value()));
354  const void* attributesAddress = static_cast<const void*>(&attributes);
355 
356  // clang-format off
357  Variable<T, R, M> res = {
358  name,
359  long_name,
360  scrubbed_spec,
361  store,
362  attrs
363  };
364  // clang-format on
365  return res;
366 }

385 template <typename T = void, DimensionIndex R = dynamic_rank,
386  ReadWriteMode M = ReadWriteMode::dynamic, typename... Option>
387 Future<Variable<T, R, M>> CreateVariable(const nlohmann::json& json_spec,
388  const nlohmann::json& json_var,
389  Option&&... options) {
390  auto spec = tensorstore::MakeReadyFuture<::nlohmann::json>(json_var);
391 
392  // FIXME - add schematized validation of the variable.
393  if (json_var.empty()) {
394  return absl::Status(absl::StatusCode::kInvalidArgument,
395  "Expected attributes to be non-empty.");
396  }
397 
398  if (!json_spec.contains("metadata")) {
399  return absl::Status(absl::StatusCode::kInvalidArgument,
400  "Variable spec requires metadata");
401  }
402 
403  if (!json_spec["metadata"].contains("dtype")) {
404  return absl::Status(absl::StatusCode::kInvalidArgument,
405  "Variable metadata requires dtype");
406  }
407 
408  MDIO_ASSIGN_OR_RETURN(auto zarr_dtype, tensorstore::internal_zarr::ParseDType(
409  json_spec["metadata"]["dtype"]))
410 
411  // Handles the use case of creating a struct array, but intending to open as
412  // void.
413  auto do_handle_structarray =
414  zarr_dtype.has_fields && !json_spec.contains("field");
415  auto json_spec_with_field = json_spec;
416  if (do_handle_structarray) {
417  // pick the first name, it won't effect the .zarray json:
418  json_spec_with_field["field"] = zarr_dtype.fields[0].name;
419  }
420 
421  auto json_spec_without_metadata = json_spec;
422  json_spec_without_metadata.erase("metadata");
423  auto future_json_store = tensorstore::MakeReadyFuture<::nlohmann::json>(
424  json_spec_without_metadata);
425 
426  auto publish = [](const ::nlohmann::json& json_var, bool isCloudStore,
427  const tensorstore::TensorStore<T, R, M>& store)
428  -> Future<tensorstore::TimestampedStorageGeneration> {
429  auto output_json = json_var;
430  output_json["_ARRAY_DIMENSIONS"] = output_json["dimension_names"];
431  output_json.erase("dimension_names");
432  output_json.erase("variable_name");
433  if (output_json.contains("metadata")) {
434  output_json["metadata"].erase("chunkGrid");
435  for (auto& item : output_json["metadata"].items()) {
436  output_json[item.key()] = std::move(item.value());
437  }
438  output_json.erase("metadata");
439  }
440  if (output_json.contains("long_name")) {
441  if (output_json["long_name"] == "") {
442  output_json.erase("long_name");
443  }
444  }
445  // Case where empty array of coordinates is provided
446  if (output_json.contains("coordinates")) {
447  auto coords = output_json["coordinates"];
448  if (coords.empty() ||
449  (coords.is_string() && coords.get<std::string>() == "")) {
450  output_json.erase("coordinates");
451  }
452  }
453  std::string outpath = "/.zattrs";
454  if (isCloudStore) {
455  outpath = ".zattrs";
456  }
457  // It's important to use the store's kvstore or else we get a race condition
458  // on "mkdir".
459  return tensorstore::kvstore::Write(store.kvstore(), outpath,
460  absl::Cord(output_json.dump(4)));
461  };
462 
463  // this is intended to handle the struct array where we "reopen" the store
464  // but this time as struct ... but we loose the capactity for forward options.
465  // FIXME - strip "create/delete existing" and foward other options.
466  auto apply_reopen = [](const tensorstore::TensorStore<T, R, M>& store,
467  const ::nlohmann::json& attributes,
468  const ::nlohmann::json& json_spec) {
469  return tensorstore::Open<T, R, M>(json_spec);
470  };
471 
472  auto build = [](const ::nlohmann::json& metadata,
473  const tensorstore::TensorStore<T, R, M>& store)
474  -> Future<Variable<T, R, M>> {
475  auto labeled_store = store;
476 
477  if (metadata.contains("dimension_names")) {
478  auto dimension_names =
479  metadata["dimension_names"].get<std::vector<std::string>>();
480  for (DimensionIndex i = 0; i < dimension_names.size(); ++i) {
481  MDIO_ASSIGN_OR_RETURN(
482  labeled_store,
483  labeled_store | tensorstore::Dims(i).Label(dimension_names[i]));
484  }
485  }
486  return from_json<T, R, M>(metadata, labeled_store);
487  };
488 
489  // Start by creating a future for the store ...
490  auto future_store = tensorstore::Open<T, R, M>(
491  json_spec_with_field, std::forward<Option>(options)...);
492 
493  auto handled_store = future_store;
494  if (do_handle_structarray) {
495  handled_store =
496  tensorstore::MapFutureValue(tensorstore::InlineExecutor{}, apply_reopen,
497  future_store, spec, future_json_store);
498  } else {
499  handled_store = std::move(future_store);
500  }
501 
502  auto variable_future = tensorstore::MapFutureValue(
503  tensorstore::InlineExecutor{}, build, spec, handled_store);
504 
505  bool isCloudStore = false;
506  std::string driver = json_spec["kvstore"]["driver"].get<std::string>();
507  if (driver == "gcs" || driver == "s3") {
508  isCloudStore = true;
509  }
510  auto isCloudStoreRF = tensorstore::MakeReadyFuture<bool>(isCloudStore);
511  auto write_metadata_future =
512  tensorstore::MapFutureValue(tensorstore::InlineExecutor{}, publish, spec,
513  isCloudStoreRF, handled_store);
514 
515  // wait for the variable to be created and the zattrs to be written.
516  auto all_done_future =
517  tensorstore::WaitAllFuture(variable_future, write_metadata_future);
518 
519  auto pair = tensorstore::PromiseFuturePair<Variable<T, R, M>>::Make();
520  all_done_future.ExecuteWhenReady(
521  [promise = pair.promise, variable_future = std::move(variable_future)](
522  tensorstore::ReadyFuture<void> readyFut) {
523  auto ready_result = readyFut.result();
524  if (!ready_result.ok()) {
525  promise.SetResult(CheckMissingDriverStatus(ready_result.status()));
526  } else {
527  promise.SetResult(variable_future.result());
528  }
529  });
530 
531  return pair.future;
532 }

545 template <typename T = void, DimensionIndex R = dynamic_rank,
546  ReadWriteMode M = ReadWriteMode::dynamic, typename... Option>
547 Future<Variable<T, R, M>> OpenVariable(const nlohmann::json& json_store,
548  Option&&... options) {
549  // Infer the name from the path
550  std::string variable_name = json_store["kvstore"]["path"].get<std::string>();
551  std::vector<std::string> pathComponents = absl::StrSplit(variable_name, "/");
552  variable_name = pathComponents.back();
553 
554  auto store_spec = json_store;
555  // retain attributes if we want to check values ...
556  ::nlohmann::json suppliedAttributes;
557  if (store_spec.contains("attributes")) {
558  suppliedAttributes["attributes"] = store_spec["attributes"];
559  store_spec.erase("attributes");
560  }
561  // attributes is not a valid key for the tensorstore open ...
562  // FIXME - resolve opening struct array with field and no metadata
563  // udpates to Tensorstore required ...
564  if (!store_spec.contains("field") && store_spec.contains("metadata")) {
565  store_spec.erase("metadata");
566  }
567  // the negative of this is valid for tensorstore ...
568 
569  auto spec = tensorstore::MakeReadyFuture<::nlohmann::json>(store_spec);
570 
571  // open a store:
572  auto future_store =
573  tensorstore::Open<T, R, M>(store_spec, std::forward<Option>(options)...);
574 
575  // start by creating a kvstore future ...
576  auto kvs_future = tensorstore::kvstore::Open(store_spec["kvstore"]);
577 
578  // go read the metadata return json ...
579  auto read = [](const tensorstore::KvStore& kvstore)
580  -> Future<tensorstore::kvstore::ReadResult> {
581  return tensorstore::kvstore::Read(kvstore, "/.zattrs");
582  };
583 
584  // go read the attributes return json ...
585  auto parse = [](const tensorstore::kvstore::ReadResult& kvs_read,
586  const ::nlohmann::json& spec) {
587  auto attributes =
588  nlohmann::json::parse(std::string(kvs_read.value), nullptr, false);
589  return attributes;
590  };
591 
592  auto make_variable = [variable_name](
593  const ::nlohmann::json& metadata,
594  const tensorstore::TensorStore<T, R, M>& store,
595  const ::nlohmann::json& suppliedAttributes)
596  -> Future<Variable<T, R, M>> {
597  // use to load the store
598  tensorstore::TensorStore<T, R, M> labeled_store = store;
599  ::nlohmann::json updated_metadata = metadata;
600 
601  // Create a new JSON object with "attributes" as the parent key
602  ::nlohmann::json new_metadata;
603  new_metadata = updated_metadata;
604  new_metadata["variable_name"] = variable_name;
605 
606  if (new_metadata.contains("_ARRAY_DIMENSIONS")) {
607  // Move "_ARRAY_DIMENSIONS" to "dimension_names"
608  new_metadata["dimension_names"] = new_metadata["_ARRAY_DIMENSIONS"];
609  new_metadata.erase("_ARRAY_DIMENSIONS");
610  }
611 
612  if (new_metadata.contains("dimension_names")) {
613  auto dimension_names =
614  new_metadata["dimension_names"].get<std::vector<std::string>>();
615  for (DimensionIndex i = 0; i < dimension_names.size(); ++i) {
616  MDIO_ASSIGN_OR_RETURN(
617  labeled_store,
618  labeled_store | tensorstore::Dims(i).Label(dimension_names[i]));
619  }
620  } else {
621  return absl::NotFoundError(
622  absl::StrCat("Field not found in JSON: ", "metadata"));
623  }
624 
625  if (new_metadata.contains("attributes")) {
626  new_metadata["metadata"]["attributes"] = new_metadata["attributes"];
627  new_metadata.erase("attributes");
628  }
629  if (new_metadata.contains("statsV1")) {
630  new_metadata["metadata"]["statsV1"] = new_metadata["statsV1"];
631  new_metadata.erase("statsV1");
632  }
633  if (new_metadata.contains("unitsV1")) {
634  new_metadata["metadata"]["unitsV1"] = new_metadata["unitsV1"];
635  new_metadata.erase("unitsV1");
636  }
637 
638  if (!suppliedAttributes.is_null()) {
639  // The supplied attributes contain some things that we do not serialize.
640  // We need to remove them. This could cause confusion. If the user
641  // specifies a different chunkGrid, it will not be used and should
642  // actually fail here.
643  nlohmann::json correctedSuppliedAttrs = suppliedAttributes;
644  if (correctedSuppliedAttrs.contains("attributes")) {
645  if (correctedSuppliedAttrs["attributes"].contains("metadata")) {
646  if (correctedSuppliedAttrs["attributes"]["metadata"].contains(
647  "chunkGrid")) {
648  correctedSuppliedAttrs["attributes"]["metadata"].erase("chunkGrid");
649  }
650  }
651  auto savedAttrs = correctedSuppliedAttrs["attributes"];
652  correctedSuppliedAttrs.erase("attributes");
653  for (auto& item : savedAttrs.items()) {
654  correctedSuppliedAttrs[item.key()] = std::move(item.value());
655  }
656  }
657  // BFS to make sure supplied attributes match stored attributes
658  nlohmann::json searchableMetadata = new_metadata;
659  if (searchableMetadata.contains("variable_name")) {
660  // Since we don't actually want to have to specify the variable name
661  searchableMetadata.erase("variable_name");
662  }
663  std::queue<std::pair<nlohmann::json, nlohmann::json>> queue;
664  queue.push({searchableMetadata, correctedSuppliedAttrs});
665  while (!queue.empty()) {
666  auto [currentMetadata, currentAttributes] = queue.front();
667  queue.pop();
668  for (auto& [key, value] : currentMetadata.items()) {
669  if (!currentAttributes.contains(key)) {
670  return absl::NotFoundError(
671  absl::StrCat("Field not found in JSON: ", key));
672  }
673  if (value.is_object() && currentAttributes[key].is_object()) {
674  // If the value is a JSON object, add it to the queue for further
675  // processing
676  queue.push({value, currentAttributes[key]});
677  } else if (value != currentAttributes[key]) {
678  return absl::InvalidArgumentError(absl::StrCat(
679  "Conflicting values for field: ", key, ". ", "Expected: ",
680  value.dump(4), ", but got: ", currentAttributes[key].dump(4)));
681  }
682  }
683  }
684  }
685 
686  return from_json<T, R, M>(new_metadata, labeled_store);
687  };
688 
689  // a future to the metadata ...
690  auto kvs_read_future = tensorstore::MapFutureValue(
691  tensorstore::InlineExecutor{}, read, kvs_future);
692 
693  auto metadata = tensorstore::MapFutureValue(tensorstore::InlineExecutor{},
694  parse, kvs_read_future, spec);
695 
696  return tensorstore::MapFutureValue(
697  tensorstore::InlineExecutor{}, make_variable, metadata, future_store,
698  tensorstore::MakeReadyFuture<::nlohmann::json>(suppliedAttributes));
699 }

711 template <typename T = void, DimensionIndex R = dynamic_rank,
712  ReadWriteMode M = ReadWriteMode::dynamic>
713 Future<Variable<T, R, M>> Open(const nlohmann::json& json_spec,
714  TransactionalOpenOptions&& options) {
715  // situations where we would create new metadata
716  if (options.open_mode == constants::kCreateClean ||
717  options.open_mode == constants::kCreate) {
718  MDIO_ASSIGN_OR_RETURN(auto json_schema, ValidateAndProcessJson(json_spec))
719  // extract the json for the store and our metadata
720  auto [json_store, metadata] = json_schema;
721  // this will write metadata
722  return CreateVariable<T, R, M>(json_store, metadata, std::move(options));
723  } else {
724  return OpenVariable<T, R, M>(json_spec, std::move(options));
725  }
726 }

946  const RangeDescriptor<Index>& desc) const {
947  auto domain = dimensions();
948  const auto labels = domain.labels();
949

1032  template <typename... Descriptors>
1033  Result<Variable> slice(const Descriptors&... descriptors) const {
1034  constexpr size_t numDescriptors = sizeof...(descriptors);
1035 
1036  auto tuple_descs = std::make_tuple(descriptors...);
1037 
1038  std::vector<DimensionIdentifier> labels;
1039  labels.reserve(numDescriptors);
1040 
1041  std::vector<Index> start, stop, step;
1042  start.reserve(numDescriptors);
1043  stop.reserve(numDescriptors);
1044  step.reserve(numDescriptors);
1045  // -1 Everything is ok
1046  // -2 Error: Step is not 1
1047  // >=0 Error: Start is greater than or equal to stop
1048  int8_t preconditionStatus = -1;
1049 
1050  std::apply(
1051  [&](const auto&... desc) {
1052  size_t idx = 0;
1053  ((
1054  [&] {
1055  if (desc.step != 1) {
1056  preconditionStatus = -2;
1057  return -2;
1058  }
1059  auto clampedDesc = sliceInRange(desc);
1060  if (clampedDesc.start > clampedDesc.stop) {
1061  preconditionStatus = idx;
1062  return 1;
1063  }
1064  if (this->hasLabel(clampedDesc.label)) {
1065  labels.push_back(clampedDesc.label);
1066  start.push_back(clampedDesc.start);
1067  stop.push_back(clampedDesc.stop);
1068  step.push_back(clampedDesc.step);
1069  }
1070  return 0; // Return a dummy value to satisfy the comma
1071  // operator
1072  }(),
1073  idx++),
1074  ...);
1075  },
1076  tuple_descs);
1077 
1078  if (preconditionStatus == -2) {
1079  return absl::InvalidArgumentError(
1080  "Only step 1 is supported for slicing.");
1081  } else if (preconditionStatus >= 0) {
1082  mdio::RangeDescriptor<Index> err;
1083  std::apply(
1084  [&](const auto&... desc) {
1085  size_t idx = 0;
1086  (([&] {
1087  if (idx == preconditionStatus) {
1088  err = desc;
1089  }
1090  idx++;
1091  }()),
1092  ...);
1093  },
1094  tuple_descs);
1095  return absl::InvalidArgumentError(
1096  std::string("Slice descriptor for ") +
1097  std::string(err.label.label()) +
1098  " had an illegal configuration.\n\tStart '" +
1099  std::to_string(err.start) + "' greater than or equal to stop '" +
1100  std::to_string(err.stop) + "'.");
1101  }
1102 
1103  auto labelSize = labels.size();
1104  if (labelSize) {
1105  std::set<std::string_view> labelSet;
1106  std::set<DimensionIndex> indexSet;
1107  for (const auto& label : labels) {
1108  labelSet.insert(label.label());
1109  indexSet.insert(label.index());
1110  }
1111 
1112  if (labelSet.size() == labelSize || indexSet.size() == labelSize) {
1113  MDIO_ASSIGN_OR_RETURN(
1114  auto slice_store,
1115  store |
1116  tensorstore::Dims(labels).HalfOpenInterval(start, stop, step));
1117  // return a new variable with the sliced store
1118  return Variable{variableName, longName, metadata, slice_store,
1119  attributes};
1120  } else if (labelSet.size() != labelSize) {
1121  // Concat the sliced Variable together if there are duplicate
1122  // labels(dimensions)
1123  std::vector<Variable> fragments;
1124  absl::Status trueStatus = absl::OkStatus();
1125  auto fragmentStore = [&](auto& descriptor) -> absl::Status {
1126  if (descriptor.label.label() == internal::kInertSliceKey) {
1127  // pass on kInertSliceKey
1128  return absl::OkStatus();
1129  }
1130  auto sliceRes = slice(descriptor);
1131  if (!sliceRes.status().ok()) {
1132  trueStatus = sliceRes.status();
1133  return trueStatus;
1134  }
1135  fragments.push_back(sliceRes.value());
1136  return absl::OkStatus();
1137  };
1138 
1139  auto status = (fragmentStore(descriptors).ok() && ...);
1140  if (!status) {
1141  return trueStatus;
1142  }
1143 
1144  if (!fragments.empty()) {
1145  // Concat appears to only work with void types, so we'll strip the
1146  // type away
1147  tensorstore::TensorStore<void, R, M> catStore;
1148  // Initialize catStore with the first fragment's store
1149  catStore = fragments.front().get_store();
1150 
1151  // Concatenate remaining fragments
1152  for (size_t i = 1; i < fragments.size(); ++i) {
1153  MDIO_ASSIGN_OR_RETURN(
1154  catStore,
1155  tensorstore::Concat({catStore, fragments[i].get_store()},
1156  /*axis=*/0));
1157  }
1158  // Recast to the original type
1159  tensorstore::TensorStore<T, R, M> typedCatStore =
1160  tensorstore::TensorStore<T, R, M>(tensorstore::unchecked,
1161  catStore);
1162  // Return a new Variable with the concatenated store
1163  return Variable{variableName, longName, metadata, typedCatStore,
1164  attributes};
1165  }
1166  return absl::InternalError("No fragments to concatenate.");
1167  }
1168 
1169  return absl::InvalidArgumentError(
1170  "Unexpected error occured while trying to slice the Variable.");
1171  }
1172  // the slice didn't change anything in the variables dimensions.
1173  return *this;
1174  }

1595  step(numDescriptors);

1599 
1600  bool preconditionStatus = true;

1606  ((

1615  return; // Exit lambda on error

1617  if (desc.step != 1) {

1619  }

1621  }(),
1622  idx++),
1623  ...);
1624  },
1625  tuple_descs);
