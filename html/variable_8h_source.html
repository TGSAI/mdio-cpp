216 absl::Status CheckMissingDriverStatus(const absl::Status& status) {
217  std::string error(status.message());
218  if (error.find("Error parsing object member \"driver\"") !=
219  std::string::npos) {
220  if (error.find("is not registered") != std::string::npos) {
221  if (error.find("gcs") != std::string::npos) {
222  return absl::InvalidArgumentError(
223  "A GCS path was detected but the GCS driver was not "
224  "registered.\nPlease ensure that your CMake includes the "
225  "mdio_INTERNAL_GCS_DRIVER_DEPS variable.");
226  } else if (error.find("s3") != std::string::npos) {
227  return absl::InvalidArgumentError(
228  "An S3 path was detected but the S3 driver was not "
229  "registered.\nPlease ensure that your CMake includes the "
230  "mdio_INTERNAL_S3_DRIVER_DEPS variable.");
231  } else {
232  return absl::InvalidArgumentError(
233  "An unexpected driver registration error has occured. Please file "
234  "a bug report with the error message to "
235  "https://github.com/TGSAI/mdio-cpp/issues\n" +
236  error);
237  }
238  }
239  }
240  return status;
241 }

259 template <ReadWriteMode Mode = ReadWriteMode::dynamic>
260 Result<std::tuple<nlohmann::json, nlohmann::json>> ValidateAndProcessJson(
261  const nlohmann::json& json_spec) {
262  // Check if attributes are in the original JSON
263  if (!json_spec.contains("attributes")) {
264  return absl::InvalidArgumentError(
265  "The json_spec does not contain 'attributes'.");
266  }
267 
268  // Check if dimensions field exists in attributes
269  if (!json_spec["attributes"].contains("dimension_names")) {
270  return absl::InvalidArgumentError(
271  "The 'attributes' does not contain 'dimension_names'.");
272  }
273 
274  nlohmann::json json_for_store = json_spec;
275 
276  // remove attributes, the v2 store won't parse it.
277  json_for_store.erase("attributes");
278 
279  // Create a new JSON and add the kvstore
280  nlohmann::json new_json = json_spec["attributes"];
281  // update the variable name
282  new_json["variable_name"] =
283  std::filesystem::path(json_spec["kvstore"]["path"]).stem().string();
284 
285  return std::make_tuple(json_for_store, new_json);
286 }

302 template <typename T = void, DimensionIndex R = dynamic_rank,
303  ReadWriteMode M = ReadWriteMode::dynamic>
304 Result<Variable<T, R, M>> from_json(
305  const ::nlohmann::json spec,
306  const tensorstore::TensorStore<T, R, M> store) {
307  auto attributes = spec;
308  // There seems to be a case where we don't actually strip out the super
309  // attributes field by this point.
310  if (attributes.contains("attributes") &&
311  attributes["attributes"].contains("variable_name")) {
312  attributes = attributes["attributes"];
313  }
314 
315  if (!attributes.contains("variable_name")) {
316  return absl::NotFoundError("Could not find Variable's name.");
317  }
318  std::string name = attributes["variable_name"];
319 
320  nlohmann::json scrubbed_spec = attributes;
321  scrubbed_spec.erase("variable_name");
322  auto specWithoutChunkgrid = spec;
323  if (specWithoutChunkgrid.contains("metadata")) {
324  if (specWithoutChunkgrid["metadata"].contains("chunkGrid")) {
325  specWithoutChunkgrid["metadata"].erase("chunkGrid");
326  }
327  }
328  auto attrsRes = UserAttributes::FromVariableJson(specWithoutChunkgrid);
329  if (!attrsRes.ok()) {
330  return attrsRes.status();
331  }
332 
333  // Default case.
334  // TODO(BrianMichell): Look into making this optional.
335  std::string long_name = "";
336  if (scrubbed_spec.contains("long_name")) {
337  if (scrubbed_spec["long_name"].is_string() &&
338  scrubbed_spec["long_name"].get<std::string>().size() > 0) {
339  long_name = scrubbed_spec["long_name"].get<std::string>();
340  } else {
341  scrubbed_spec.erase("long_name");
342  }
343  } // Not having this field is fine. If it doesn't get seralized returning an
344  // error status is not necessary.
345 
346  // These live in our `UserAttributes` object and are technically mutable. The
347  // best kind of mutable!
348  if (scrubbed_spec.contains("metadata")) {
349  if (scrubbed_spec["metadata"].contains("attributes")) {
350  scrubbed_spec["metadata"].erase("attributes");
351  }
352  if (scrubbed_spec["metadata"].contains("statsV1")) {
353  scrubbed_spec["metadata"].erase("statsV1");
354  }
355  if (scrubbed_spec["metadata"].contains("unitsV1")) {
356  scrubbed_spec["metadata"].erase("unitsV1");
357  }
358  }
359 
360  std::shared_ptr<std::shared_ptr<UserAttributes>> attrs =
361  std::make_shared<std::shared_ptr<UserAttributes>>(
362  std::make_shared<UserAttributes>(attrsRes.value()));
363  const void* attributesAddress = static_cast<const void*>(&attributes);
364 
365  // clang-format off
366  Variable<T, R, M> res = {
367  name,
368  long_name,
369  scrubbed_spec,
370  store,
371  attrs
372  };
373  // clang-format on
374  return res;
375 }

394 template <typename T = void, DimensionIndex R = dynamic_rank,
395  ReadWriteMode M = ReadWriteMode::dynamic, typename... Option>
396 Future<Variable<T, R, M>> CreateVariable(const nlohmann::json& json_spec,
397  const nlohmann::json& json_var,
398  Option&&... options) {
399  auto spec = tensorstore::MakeReadyFuture<::nlohmann::json>(json_var);
400 
401  // FIXME - add schematized validation of the variable.
402  if (json_var.empty()) {
403  return absl::Status(absl::StatusCode::kInvalidArgument,
404  "Expected attributes to be non-empty.");
405  }
406 
407  if (!json_spec.contains("metadata")) {
408  return absl::Status(absl::StatusCode::kInvalidArgument,
409  "Variable spec requires metadata");
410  }
411 
412  if (!json_spec["metadata"].contains("dtype")) {
413  return absl::Status(absl::StatusCode::kInvalidArgument,
414  "Variable metadata requires dtype");
415  }
416 
417  MDIO_ASSIGN_OR_RETURN(auto zarr_dtype, tensorstore::internal_zarr::ParseDType(
418  json_spec["metadata"]["dtype"]))
419 
420  // Handles the use case of creating a struct array, but intending to open as
421  // void.
422  auto do_handle_structarray =
423  zarr_dtype.has_fields && !json_spec.contains("field");
424  auto json_spec_with_field = json_spec;
425  if (do_handle_structarray) {
426  // pick the first name, it won't effect the .zarray json:
427  json_spec_with_field["field"] = zarr_dtype.fields[0].name;
428  }
429 
430  auto json_spec_without_metadata = json_spec;
431  json_spec_without_metadata.erase("metadata");
432  auto future_json_store = tensorstore::MakeReadyFuture<::nlohmann::json>(
433  json_spec_without_metadata);
434 
435  auto publish = [](const ::nlohmann::json& json_var, bool isCloudStore,
436  const tensorstore::TensorStore<T, R, M>& store)
437  -> Future<tensorstore::TimestampedStorageGeneration> {
438  auto output_json = json_var;
439  output_json["_ARRAY_DIMENSIONS"] = output_json["dimension_names"];
440  output_json.erase("dimension_names");
441  output_json.erase("variable_name");
442  if (output_json.contains("metadata")) {
443  output_json["metadata"].erase("chunkGrid");
444  for (auto& item : output_json["metadata"].items()) {
445  output_json[item.key()] = std::move(item.value());
446  }
447  output_json.erase("metadata");
448  }
449  if (output_json.contains("long_name")) {
450  if (output_json["long_name"] == "") {
451  output_json.erase("long_name");
452  }
453  }
454  // Case where empty array of coordinates is provided
455  if (output_json.contains("coordinates")) {
456  auto coords = output_json["coordinates"];
457  if (coords.empty() ||
458  (coords.is_string() && coords.get<std::string>() == "")) {
459  output_json.erase("coordinates");
460  }
461  }
462  std::string outpath = "/.zattrs";
463  if (isCloudStore) {
464  outpath = ".zattrs";
465  }
466  // It's important to use the store's kvstore or else we get a race condition
467  // on "mkdir".
468  return tensorstore::kvstore::Write(store.kvstore(), outpath,
469  absl::Cord(output_json.dump(4)));
470  };
471 
472  // this is intended to handle the struct array where we "reopen" the store
473  // but this time as struct ... but we loose the capactity for forward options.
474  // FIXME - strip "create/delete existing" and foward other options.
475  auto apply_reopen = [](const tensorstore::TensorStore<T, R, M>& store,
476  const ::nlohmann::json& attributes,
477  const ::nlohmann::json& json_spec) {
478  return tensorstore::Open<T, R, M>(json_spec);
479  };
480 
481  auto build = [](const ::nlohmann::json& metadata,
482  const tensorstore::TensorStore<T, R, M>& store)
483  -> Future<Variable<T, R, M>> {
484  auto labeled_store = store;
485 
486  if (metadata.contains("dimension_names")) {
487  auto dimension_names =
488  metadata["dimension_names"].get<std::vector<std::string>>();
489  for (DimensionIndex i = 0; i < dimension_names.size(); ++i) {
490  MDIO_ASSIGN_OR_RETURN(
491  labeled_store,
492  labeled_store | tensorstore::Dims(i).Label(dimension_names[i]));
493  }
494  }
495  return from_json<T, R, M>(metadata, labeled_store);
496  };
497 
498  // Start by creating a future for the store ...
499  auto future_store = tensorstore::Open<T, R, M>(
500  json_spec_with_field, std::forward<Option>(options)...);
501 
502  auto handled_store = future_store;
503  if (do_handle_structarray) {
504  handled_store =
505  tensorstore::MapFutureValue(tensorstore::InlineExecutor{}, apply_reopen,
506  future_store, spec, future_json_store);
507  } else {
508  handled_store = std::move(future_store);
509  }
510 
511  auto variable_future = tensorstore::MapFutureValue(
512  tensorstore::InlineExecutor{}, build, spec, handled_store);
513 
514  bool isCloudStore = false;
515  std::string driver = json_spec["kvstore"]["driver"].get<std::string>();
516  if (driver == "gcs" || driver == "s3") {
517  isCloudStore = true;
518  }
519  auto isCloudStoreRF = tensorstore::MakeReadyFuture<bool>(isCloudStore);
520  auto write_metadata_future =
521  tensorstore::MapFutureValue(tensorstore::InlineExecutor{}, publish, spec,
522  isCloudStoreRF, handled_store);
523 
524  // wait for the variable to be created and the zattrs to be written.
525  auto all_done_future =
526  tensorstore::WaitAllFuture(variable_future, write_metadata_future);
527 
528  auto pair = tensorstore::PromiseFuturePair<Variable<T, R, M>>::Make();
529  all_done_future.ExecuteWhenReady(
530  [promise = pair.promise, variable_future = std::move(variable_future)](
531  tensorstore::ReadyFuture<void> readyFut) {
532  auto ready_result = readyFut.result();
533  if (!ready_result.ok()) {
534  promise.SetResult(CheckMissingDriverStatus(ready_result.status()));
535  } else {
536  promise.SetResult(variable_future.result());
537  }
538  });
539 
540  return pair.future;
541 }

554 template <typename T = void, DimensionIndex R = dynamic_rank,
555  ReadWriteMode M = ReadWriteMode::dynamic, typename... Option>
556 Future<Variable<T, R, M>> OpenVariable(const nlohmann::json& json_store,
557  Option&&... options) {
558  // Infer the name from the path
559  std::string variable_name = json_store["kvstore"]["path"].get<std::string>();
560  std::vector<std::string> pathComponents = absl::StrSplit(variable_name, "/");
561  variable_name = pathComponents.back();
562 
563  auto store_spec = json_store;
564  // retain attributes if we want to check values ...
565  ::nlohmann::json suppliedAttributes;
566  if (store_spec.contains("attributes")) {
567  suppliedAttributes["attributes"] = store_spec["attributes"];
568  store_spec.erase("attributes");
569  }
570  // attributes is not a valid key for the tensorstore open ...
571  // FIXME - resolve opening struct array with field and no metadata
572  // updates to Tensorstore required ...
573  if (!store_spec.contains("field") && store_spec.contains("metadata")) {
574  store_spec.erase("metadata");
575  }
576  // the negative of this is valid for tensorstore ...
577 
578  auto spec = tensorstore::MakeReadyFuture<::nlohmann::json>(store_spec);
579 
580  // open a store:
581  auto future_store =
582  tensorstore::Open<T, R, M>(store_spec, std::forward<Option>(options)...);
583 
584  // start by creating a kvstore future ...
585  auto kvs_future = tensorstore::kvstore::Open(store_spec["kvstore"]);
586 
587  // go read the metadata return json ...
588  auto read = [](const tensorstore::KvStore& kvstore)
589  -> Future<tensorstore::kvstore::ReadResult> {
590  return tensorstore::kvstore::Read(kvstore, "/.zattrs");
591  };
592 
593  // go read the attributes return json ...
594  auto parse = [](const tensorstore::kvstore::ReadResult& kvs_read,
595  const ::nlohmann::json& spec) {
596  auto attributes =
597  nlohmann::json::parse(std::string(kvs_read.value), nullptr, false);
598  return attributes;
599  };

601  auto make_variable = [variable_name](
602  const ::nlohmann::json& metadata,
603  const tensorstore::TensorStore<T, R, M>& store,
604  const ::nlohmann::json& suppliedAttributes)
605  -> Future<Variable<T, R, M>> {
606  // use to load the store
607  tensorstore::TensorStore<T, R, M> labeled_store = store;
608  ::nlohmann::json updated_metadata = metadata;
609 
610  // Create a new JSON object with "attributes" as the parent key
611  ::nlohmann::json new_metadata;
612  new_metadata = updated_metadata;
613  new_metadata["variable_name"] = variable_name;
614 
615  if (new_metadata.contains("_ARRAY_DIMENSIONS")) {
616  // Move "_ARRAY_DIMENSIONS" to "dimension_names"
617  new_metadata["dimension_names"] = new_metadata["_ARRAY_DIMENSIONS"];
618  new_metadata.erase("_ARRAY_DIMENSIONS");
619  }
620 
621  if (new_metadata.contains("dimension_names")) {
622  auto dimension_names =
623  new_metadata["dimension_names"].get<std::vector<std::string>>();
624  for (DimensionIndex i = 0; i < dimension_names.size(); ++i) {
625  MDIO_ASSIGN_OR_RETURN(
626  labeled_store,
627  labeled_store | tensorstore::Dims(i).Label(dimension_names[i]));
628  }
629  } else {
630  return absl::NotFoundError(
631  absl::StrCat("Field not found in JSON: ", "metadata"));
632  }
633 
634  if (new_metadata.contains("attributes")) {
635  new_metadata["metadata"]["attributes"] = new_metadata["attributes"];
636  new_metadata.erase("attributes");
637  }
638  if (new_metadata.contains("statsV1")) {
639  new_metadata["metadata"]["statsV1"] = new_metadata["statsV1"];
640  new_metadata.erase("statsV1");
641  }
642  if (new_metadata.contains("unitsV1")) {
643  new_metadata["metadata"]["unitsV1"] = new_metadata["unitsV1"];
644  new_metadata.erase("unitsV1");
645  }
646 
647  if (!suppliedAttributes.is_null()) {
648  // The supplied attributes contain some things that we do not serialize.
649  // We need to remove them. This could cause confusion. If the user
650  // specifies a different chunkGrid, it will not be used and should
651  // actually fail here.
652  nlohmann::json correctedSuppliedAttrs = suppliedAttributes;
653  if (correctedSuppliedAttrs.contains("attributes")) {
654  if (correctedSuppliedAttrs["attributes"].contains("metadata")) {
655  if (correctedSuppliedAttrs["attributes"]["metadata"].contains(
656  "chunkGrid")) {
657  correctedSuppliedAttrs["attributes"]["metadata"].erase("chunkGrid");
658  }
659  }
660  auto savedAttrs = correctedSuppliedAttrs["attributes"];
661  correctedSuppliedAttrs.erase("attributes");
662  for (auto& item : savedAttrs.items()) {
663  correctedSuppliedAttrs[item.key()] = std::move(item.value());
664  }
665  }
666  // BFS to make sure supplied attributes match stored attributes
667  nlohmann::json searchableMetadata = new_metadata;
668  if (searchableMetadata.contains("variable_name")) {
669  // Since we don't actually want to have to specify the variable name
670  searchableMetadata.erase("variable_name");
671  }
672  std::queue<std::pair<nlohmann::json, nlohmann::json>> queue;
673  queue.push({searchableMetadata, correctedSuppliedAttrs});
674  while (!queue.empty()) {
675  auto [currentMetadata, currentAttributes] = queue.front();
676  queue.pop();
677  for (auto& [key, value] : currentMetadata.items()) {
678  if (!currentAttributes.contains(key)) {
679  return absl::NotFoundError(
680  absl::StrCat("Field not found in JSON: ", key));
681  }
682  if (value.is_object() && currentAttributes[key].is_object()) {
683  // If the value is a JSON object, add it to the queue for further
684  // processing
685  queue.push({value, currentAttributes[key]});
686  } else if (value != currentAttributes[key]) {
687  return absl::InvalidArgumentError(absl::StrCat(
688  "Conflicting values for field: ", key, ". ", "Expected: ",
689  value.dump(4), ", but got: ", currentAttributes[key].dump(4)));
690  }
691  }
692  }
693  }
694 
695  return from_json<T, R, M>(new_metadata, labeled_store);
696  };
697 
698  // a future to the metadata ...
699  auto kvs_read_future = tensorstore::MapFutureValue(
700  tensorstore::InlineExecutor{}, read, kvs_future);
701 
702  auto metadata = tensorstore::MapFutureValue(tensorstore::InlineExecutor{},
703  parse, kvs_read_future, spec);
704 
705  return tensorstore::MapFutureValue(
706  tensorstore::InlineExecutor{}, make_variable, metadata, future_store,
707  tensorstore::MakeReadyFuture<::nlohmann::json>(suppliedAttributes));
708 }

720 template <typename T = void, DimensionIndex R = dynamic_rank,
721  ReadWriteMode M = ReadWriteMode::dynamic>
722 Future<Variable<T, R, M>> Open(const nlohmann::json& json_spec,
723  TransactionalOpenOptions&& options) {
724  // situations where we would create new metadata
725  if (options.open_mode == constants::kCreateClean ||
726  options.open_mode == constants::kCreate) {
727  MDIO_ASSIGN_OR_RETURN(auto json_schema, ValidateAndProcessJson(json_spec))
728  // extract the json for the store and our metadata
729  auto [json_store, metadata] = json_schema;
730  // this will write metadata
731  return CreateVariable<T, R, M>(json_store, metadata, std::move(options));
732  } else {
733  return OpenVariable<T, R, M>(json_spec, std::move(options));
734  }
735 }

956  RangeDescriptor<Index> sliceInRange(
957  const RangeDescriptor<Index>& desc) const {
958  auto domain = dimensions();
959  const auto labels = domain.labels();
960 
961  for (size_t idx = 0; idx < labels.size(); ++idx) {
962  if (labels[idx] == desc.label) {

964  return {desc.label, // label
965  desc.start < domain.origin()[idx] ? domain.origin()[idx]

968  : desc.stop, // stop

973  // valid

1043  template <typename... Descriptors>
1044  Result<Variable> slice(const Descriptors&... descriptors) const {
1045  constexpr size_t numDescriptors = sizeof...(descriptors);
1046 
1047  auto tuple_descs = std::make_tuple(descriptors...);
1048 
1049  std::vector<DimensionIdentifier> labels;
1050  labels.reserve(numDescriptors);
1051 
1052  std::vector<Index> start, stop, step;
1053  start.reserve(numDescriptors);
1054  stop.reserve(numDescriptors);
1055  step.reserve(numDescriptors);
1056  // -1 Everything is ok
1057  // -2 Error: Step is not 1
1058  // >=0 Error: Start is greater than or equal to stop
1059  int8_t preconditionStatus = -1;
1060 
1061  std::apply(
1062  [&](const auto&... desc) {
1063  size_t idx = 0;
1064  ((
1065  [&] {
1066  if (desc.step != 1) {
1067  preconditionStatus = -2;
1068  return -2;
1069  }
1070  auto clampedDesc = sliceInRange(desc);
1071  if (clampedDesc.start > clampedDesc.stop) {
1072  preconditionStatus = idx;
1073  return 1;
1074  }
1075  if (this->hasLabel(clampedDesc.label)) {
1076  labels.push_back(clampedDesc.label);
1077  start.push_back(clampedDesc.start);
1078  stop.push_back(clampedDesc.stop);
1079  step.push_back(clampedDesc.step);
1080  }
1081  return 0; // Return a dummy value to satisfy the comma
1082  // operator
1083  }(),
1084  idx++),
1085  ...);
1086  },
1087  tuple_descs);
1088 
1089  if (preconditionStatus == -2) {
1090  return absl::InvalidArgumentError(
1091  "Only step 1 is supported for slicing.");
1092  } else if (preconditionStatus >= 0) {
1093  mdio::RangeDescriptor<Index> err;
1094  std::apply(
1095  [&](const auto&... desc) {
1096  size_t idx = 0;
1097  (([&] {
1098  if (idx == preconditionStatus) {
1099  err = desc;
1100  }
1101  idx++;
1102  }()),
1103  ...);
1104  },
1105  tuple_descs);
1106  return absl::InvalidArgumentError(
1107  std::string("Slice descriptor for ") +
1108  std::string(err.label.label()) +
1109  " had an illegal configuration.\n\tStart '" +
1110  std::to_string(err.start) + "' greater than or equal to stop '" +
1111  std::to_string(err.stop) + "'.");
1112  }
1113 
1114  auto labelSize = labels.size();
1115  if (labelSize) {
1116  std::set<std::string_view> labelSet;
1117  std::set<DimensionIndex> indexSet;
1118  for (const auto& label : labels) {
1119  labelSet.insert(label.label());
1120  indexSet.insert(label.index());
1121  }
1122 
1123  if (labelSet.size() == labelSize || indexSet.size() == labelSize) {
1124  MDIO_ASSIGN_OR_RETURN(
1125  auto slice_store,
1126  store |
1127  tensorstore::Dims(labels).HalfOpenInterval(start, stop, step));
1128  // return a new variable with the sliced store
1129  return Variable{variableName, longName, metadata, slice_store,
1130  attributes};
1131  } else if (labelSet.size() != labelSize) {
1132  // Concat the sliced Variable together if there are duplicate
1133  // labels(dimensions)
1134  std::vector<Variable> fragments;
1135  absl::Status trueStatus = absl::OkStatus();
1136  auto fragmentStore = [&](auto& descriptor) -> absl::Status {
1137  if (descriptor.label.label() == internal::kInertSliceKey) {
1138  // pass on kInertSliceKey
1139  return absl::OkStatus();
1140  }
1141  auto sliceRes = slice(descriptor);
1142  if (!sliceRes.status().ok()) {
1143  trueStatus = sliceRes.status();
1144  return trueStatus;
1145  }
1146  fragments.push_back(sliceRes.value());
1147  return absl::OkStatus();
1148  };
1149 
1150  auto status = (fragmentStore(descriptors).ok() && ...);
1151  if (!status) {
1152  return trueStatus;
1153  }
1154 
1155  if (!fragments.empty()) {
1156  // Concat appears to only work with void types, so we'll strip the
1157  // type away
1158  tensorstore::TensorStore<void, R, M> catStore;
1159  // Initialize catStore with the first fragment's store
1160  catStore = fragments.front().get_store();
1161 
1162  // Concatenate remaining fragments
1163  for (size_t i = 1; i < fragments.size(); ++i) {
1164  MDIO_ASSIGN_OR_RETURN(
1165  catStore,
1166  tensorstore::Concat({catStore, fragments[i].get_store()},
1167  /*axis=*/0));
1168  }
1169  // Recast to the original type
1170  tensorstore::TensorStore<T, R, M> typedCatStore =
1171  tensorstore::TensorStore<T, R, M>(tensorstore::unchecked,
1172  catStore);
1173  // Return a new Variable with the concatenated store
1174  return Variable{variableName, longName, metadata, typedCatStore,
1175  attributes};
1176  }
1177  return absl::InternalError("No fragments to concatenate.");
1178  }
1179 
1180  return absl::InvalidArgumentError(
1181  "Unexpected error occured while trying to slice the Variable.");
1182  }
1183  // the slice didn't change anything in the variables dimensions.
1184  return *this;
1185  }

1856  // data type as "void", and the internal data type of the array according to
1857  // the variable

1859  variable.get_store().domain().box(), mdio::ContiguousLayoutOrder::c,
1860  tensorstore::value_init, variable.dtype());
1861

1863  variable.dtype() == constants::kFloat64) {

1865  size_t num_elements = variable.num_samples();
1866  size_t element_size = variable.dtype().size();
1867 
1868  if (variable.dtype() == constants::kFloat32) {
1869  auto* data = reinterpret_cast<float*>(_array.data());
1870  std::fill_n(data, num_elements, std::numeric_limits<float>::quiet_NaN());
1871  } else { // double
1872  auto* data = reinterpret_cast<double*>(_array.data());
1873  std::fill_n(data, num_elements, std::numeric_limits<double>::quiet_NaN());
1874  }
1875  }

1878  // templated. this can fail if the types are inconsistent, at which point it
1879  // will return with a status.

1881  tensorstore::StaticDataTypeCast<T>(std::move(_array)))
1882

1884
