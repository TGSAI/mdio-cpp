78 template <typename TraceType, typename CoordinateType>
79 mdio::Result<VolumeStats<TraceType, CoordinateType>> CalculateVolumeStatistics(mdio::Dataset& ds) {
80 
81  // We could get the chunk sizes from the Variable objects, but for simplicity, we will just use these constants.
82  // See the Variable::get_chunk_shape() method for more information.
83  const mdio::Index INLINE_CHUNK_SIZE = 128;
84  const mdio::Index CROSSLINE_CHUNK_SIZE = 128;
85  // These are arbitrary indices that align with the beginning of a chunk.
86  const mdio::Index INLINE_START = 512;
87  const mdio::Index CROSSLINE_START = 256;
88 
89  // Initialize the statistics object.
90  VolumeStats<TraceType, CoordinateType> stats = {
91  .avg_amplitude = 0,
92  .max_amplitude = std::numeric_limits<TraceType>::min(),
93  .min_amplitude = std::numeric_limits<TraceType>::max(),
94  .peak_amplitude_inline = 0,
95  .peak_amplitude_crossline = 0,
96  .trough_amplitude_inline = 0,
97  .trough_amplitude_crossline = 0,
98  .num_samples = 0
99  };
100 
101  // First, we are going to enter some arbitrary inline and crossline indices that align with the beginning of a chunk.
102  mdio::RangeDescriptor<> inlineDesc = {"inline", INLINE_START, INLINE_START + INLINE_CHUNK_SIZE, 1};
103  mdio::RangeDescriptor<> crosslineDesc = {"crossline", CROSSLINE_START, CROSSLINE_START + CROSSLINE_CHUNK_SIZE, 1};
104  // We will not slice the time dimension, as we want the full trace length.
105 
106  // We will set up these range descriptors to hold our local indices as we iterate inside our subset dataset.
107  mdio::RangeDescriptor<> inlineIndexDesc = {"inline", 0, INLINE_CHUNK_SIZE, 1};
108  mdio::RangeDescriptor<> crosslineIndexDesc = {"crossline", 0, CROSSLINE_CHUNK_SIZE, 1};
109 
110  // Iterate over the inline and crossline chunks.
111  for (mdio::Index inline_idx = INLINE_START; inline_idx<INLINE_START + INLINE_CHUNK_SIZE; inline_idx += INLINE_CHUNK_SIZE) {
112  for (mdio::Index crossline_idx = CROSSLINE_START; crossline_idx<CROSSLINE_START + CROSSLINE_CHUNK_SIZE; crossline_idx += CROSSLINE_CHUNK_SIZE) {
113 
114  // Set up the slicing descriptors with the current global chunk to process.
115  inlineDesc.start = inline_idx;
116  inlineDesc.stop = inline_idx + INLINE_CHUNK_SIZE;
117  crosslineDesc.start = crossline_idx;
118  crosslineDesc.stop = crossline_idx + CROSSLINE_CHUNK_SIZE;
119 
120  // Ensure our local indices are reset to the start of the chunk.
121  inlineIndexDesc.start = 0;
122  crosslineIndexDesc.start = 0;
123 
124  // The result of the isel operation is a new Dataset object with the index extents we described above.
125  MDIO_ASSIGN_OR_RETURN(auto sliced_ds, ds.isel(inlineDesc, crosslineDesc));
126 
127  // Now we get the relevant Variables from the sliced dataset to do our statistics.
128  MDIO_ASSIGN_OR_RETURN(auto traceVariable, sliced_ds.variables.get<TraceType>("seismic"));
129  MDIO_ASSIGN_OR_RETURN(auto inlineVariable, sliced_ds.variables.get<CoordinateType>("inline"));
130  MDIO_ASSIGN_OR_RETURN(auto crosslineVariable, sliced_ds.variables.get<CoordinateType>("crossline"));
131  MDIO_ASSIGN_OR_RETURN(auto timeVariable, sliced_ds.variables.get<CoordinateType>("time"));
132 
133  // Read is an asynchronous operation which executes immediately and with multiple threads and processes.
134  auto traceFuture = traceVariable.Read();
135  auto inlineFuture = inlineVariable.Read();
136  auto crosslineFuture = crosslineVariable.Read();
137  // We don't need to read the time Variable, we just get it from the sliced dataset to get the trace length.
138 
139  // Calling the value() and status() methods are blocking operations that wait for the future to complete and resolve.
140  if (!traceFuture.status().ok()) {
141  return traceFuture.status();
142  }
143 
144  // We only need to wait for the trace to resolve, as the coordinates won't be needed until after the first trace is processed.
145  auto trace = traceFuture.value(); // Resolve the future into a strongly typed VariableData object.
146  auto tracePtr = trace.get_data_accessor().data(); // Get a pointer to the flattened trace data.
147  auto traceOffset = trace.get_flattened_offset(); // The flattened trace data may have some constant pointer offset at the start.
148  auto traceLength = timeVariable.num_samples(); // Notice, we used the time Variable to get the trace length.
149 
150  auto numTraceSamples = trace.num_samples(); // This gives us the total number of samples in the chunk.
151 
152  // Now, we are going to process the data tracewise.
153  for (mdio::Index traceIdx = 0; traceIdx < numTraceSamples; traceIdx += traceLength) {
154  auto local_minmax = std::minmax_element(tracePtr + traceOffset + traceIdx, tracePtr + traceOffset + traceIdx + traceLength);
155  auto local_min = *(local_minmax.first);
156  auto local_max = *(local_minmax.second);
157 
158  // Calculate the average amplitude for this trace.
159  auto local_avg = std::accumulate(tracePtr + traceOffset + traceIdx, tracePtr + traceOffset + traceIdx + traceLength, 0.0) / traceLength;
160  stats.avg_amplitude += local_avg;
161 
162  if (local_max > stats.max_amplitude) {
163  stats.max_amplitude = local_max;
164  MDIO_ASSIGN_OR_RETURN(auto coords, GetCoordinatePair(inlineFuture, crosslineFuture, inlineIndexDesc.start, crosslineIndexDesc.start));
165  stats.peak_amplitude_inline = std::get<0>(coords);
166  stats.peak_amplitude_crossline = std::get<1>(coords);
167  }
168  if (local_min < stats.min_amplitude) {
169  stats.min_amplitude = local_min;
170  MDIO_ASSIGN_OR_RETURN(auto coords, GetCoordinatePair(inlineFuture, crosslineFuture, inlineIndexDesc.start, crosslineIndexDesc.start));
171  stats.trough_amplitude_inline = std::get<0>(coords);
172  stats.trough_amplitude_crossline = std::get<1>(coords);
173  }
174 
175  // Increment the local indices of the chunk.
176  inlineIndexDesc.start += inlineIndexDesc.step;
177  if (inlineIndexDesc.start >= inlineIndexDesc.stop) {
178  inlineIndexDesc.start = 0;
179  crosslineIndexDesc.start += crosslineIndexDesc.step;
180  if (crosslineIndexDesc.start >= crosslineIndexDesc.stop) {
181  crosslineIndexDesc.start = 0;
182  }
183  }
184  }
185  stats.num_samples += numTraceSamples;
186  }
187  }
188  stats.avg_amplitude /= stats.num_samples;
189  return stats;
190 }

201 template <typename CoordinateType>
202 mdio::Result<std::tuple<CoordinateType, CoordinateType>> GetCoordinatePair(mdio::Future<mdio::VariableData<CoordinateType>>& inlineFuture, mdio::Future<mdio::VariableData<CoordinateType>>& crosslineFuture, const mdio::Index inline_idx, const mdio::Index crossline_idx) {
203  // Ensure that the futures are able to be resolved without error.
204  if (!inlineFuture.status().ok()) {
205  return inlineFuture.status();
206  }
207  if (!crosslineFuture.status().ok()) {
208  return crosslineFuture.status();
209  }
210  // Resolve the futures into VariableData objects
211  auto inlineData = inlineFuture.value();
212  auto crosslineData = crosslineFuture.value();
213  // Get the pointers to their data
214  auto inlinePtr = inlineData.get_data_accessor().data();
215  auto crosslinePtr = crosslineData.get_data_accessor().data();
216  // They will likely have some pointer offset before the actual data starts, so we need to add that to the index.
217  auto inlineOffset = inlineData.get_flattened_offset();
218  auto crosslineOffset = crosslineData.get_flattened_offset();
219  // Build the tuple of the inline and crossline coordinate values and return it.
220  return std::make_tuple(inlinePtr[inlineOffset + inline_idx], crosslinePtr[crosslineOffset + crossline_idx]);
221 }
