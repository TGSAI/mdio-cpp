57 template <typename T>
58 std::tuple<T, T> GetExtents(mdio::VariableData<T>& var) {
59  auto dataPtr = var.get_data_accessor().data();
60  auto offset = var.get_flattened_offset();
61  auto num_samples = var.num_samples();
62 
63  auto result = std::minmax_element(dataPtr + offset, dataPtr + offset + num_samples);
64  return std::make_tuple(*(result.first), *(result.second));
65 }
